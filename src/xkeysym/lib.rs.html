178    /// Translate a keyboard symbol to its approximate character.
179    ///
180    /// The translation follows the `xkb_keysym_to_utf32` function.
181    pub fn key_char(self) -> Option<char> {
182        let keysym = self.0;
183
184        // First check for Latin-1 characters (1:1 mapping).
185        if matches!(keysym, 0x0020..=0x007e) || matches!(keysym, 0x00a0..=0x00ff) {
186            return char::from_u32(keysym);
187        }
188
189        // Patch encoding botch.
190        if keysym == key::KP_Space {
191            return char::from_u32(key::space & 0x7f);
192        }
193
194        // Special keysyms.
195        if matches!(keysym, key::BackSpace..=key::Clear)
196            || matches!(keysym, key::KP_Multiply..=key::KP_9)
197            || keysym == key::Return
198            || keysym == key::Escape
199            || keysym == key::Delete
200            || keysym == key::KP_Tab
201            || keysym == key::KP_Enter
202            || keysym == key::KP_Equal
203        {
204            return char::from_u32(keysym & 0x7f);
205        }
206
207        // Also check for directly encoded unicode codepoints.
208
209        // Exclude surrogates: they are invalid in UTF-32.
210        // See https://www.unicode.org/versions/Unicode15.0.0/ch03.pdf#G28875
211        // for further details.
212        if matches!(keysym, 0x0100d800..=0x0100dfff) {
213            return None;
214        }
215
216        // In theory, this is supposed to start from 0x100100, such that the ASCII
217        // range, which is already covered by 0x00-0xff, can't be encoded in two
218        // ways. However, changing this after a couple of decades probably won't
219        // go well, so it stays as it is.
220        if matches!(keysym, 0x01000000..=0x0110ffff) {
221            return char::from_u32(keysym - 0x01000000);
222        }
223
224        // Search the main table.
225        KEYSYMTAB
226            .binary_search_by_key(&keysym, |pair| pair.keysym as u32)
227            .ok()
228            .and_then(|idx| char::from_u32(KEYSYMTAB[idx].ucs as u32))
229    }
230
231    /// Create a [`Keysym`] from the given `char`.
232    ///
233    /// This function replecates the `xkb_utf32_to_keysym` behavior.
234    pub fn from_char(ch: char) -> Self {
235        let ucs = ch as u32;
236
237        // First check for Latin-1 characters (1:1 mapping).
238        if matches!(ucs, 0x0020..=0x007e) || matches!(ucs, 0x00a0..=0x00ff) {
239            return Self::new(ucs);
240        }
241
242        // Special keysyms.
243        if (key::BackSpace & 0x7f..=key::Clear & 0x7f).contains(&ucs)
244            || ucs == key::Return & 0x7f
245            || ucs == key::Escape & 0x7f
246        {
247            return Self::new(ucs | 0xff00);
248        }
249
250        if ucs == key::Delete & 0x7f {
251            return Self::new(key::Delete);
252        }
253
254        // Unicode non-symbols and code points outside Unicode planes.
255        if matches!(ucs, 0xd800..=0xdfff)
256            || matches!(ucs, 0xfdd0..=0xfdef)
257            || ucs > 0x10ffff
258            || (ucs & 0xfffe == 0xfffe)
259        {
260            return Keysym::NoSymbol;
261        }
262
263        // Search main table.
264        KEYSYMTAB
265            .iter()
266            .find(|pair| pair.ucs as u32 == ucs)
267            .map_or(Self::new(ucs | 0x01000000), |pair| {
268                Self::new(pair.keysym as u32)
269            })
270    }
271}

288/// Get the keyboard symbol from a keyboard code and its column.
289///
290/// `min_keycode` can be retrieved from the X11 setup, and `keysyms_per_keycode` and `keysyms` can be
291/// retrieved from the X11 server through the `GetKeyboardMapping` request.
292pub fn keysym(
293    keycode: KeyCode,
294    mut column: u8,
295    min_keycode: KeyCode,
296    keysyms_per_keycode: u8,
297    keysyms: &[RawKeysym],
298) -> Option<Keysym> {
299    if column >= keysyms_per_keycode && column > 3 {
300        return None;
301    }
302
303    // Get the keysyms to consider.
304    let start = (keycode.0 - min_keycode.0) as usize * keysyms_per_keycode as usize;
305    let end = start + keysyms_per_keycode as usize;
306    let keysyms = &keysyms[start..end];
307
308    // See which keysym we should get.
309    let mut per = keysyms_per_keycode as usize;
310    if column < 4 {
311        // If we're going past the traditional upper/lower keys, we need to figure out where
312        // our keysym is.
313        if column >= 2 {
314            // See how many keysyms we actually have in this column.
315            loop {
316                // There will always be at least one keysym in this column.
317                if per <= 1 {
318                    break;
319                }
320
321                // If the keysym we're looking at isn't NoSymbol, we're done.
322                if keysyms[per - 1] != Keysym::NoSymbol.0 {
323                    break;
324                }
325
326                // This column isn't as big as `per`, subtract it.
327                per -= 1;
328            }
329
330            // If this keysym doesn't go past the traditional upper/lower keys, adjust column
331            // accordingly.
332            if per <= 2 {
333                column %= 2;
334            }
335        }
336
337        // Convert to upper/lower ourselves if the keysym doesn't support it.
338        let alt_column = (column | 1) as usize;
339        if per <= alt_column || keysyms[alt_column] == Keysym::NoSymbol.0 {
340            // Convert to upper/lower case.
341            let (upper, lower) = convert_case(Keysym(*keysyms.get(column as usize & !1)?));
342            return Some(if column & 1 == 0 { upper } else { lower });
343        }
344    }
345
346    keysyms.get(column as usize).map(|&keysym| Keysym(keysym))
347}

349/// Convert a keysym to its uppercase/lowercase equivalents.
350const fn convert_case(keysym: Keysym) -> (Keysym, Keysym) {
351    // by default, they're both the regular keysym
352    let (mut upper, mut lower) = (keysym.0, keysym.0);
353
354    // tell which language it belongs to
355    #[allow(non_upper_case_globals)]
356    match keysym.0 {
357        key::A..=key::Z => lower += key::a - key::A,
358        key::a..=key::z => upper -= key::a - key::A,
359        key::Agrave..=key::Odiaeresis => lower += key::agrave - key::Agrave,
360        key::agrave..=key::odiaeresis => upper -= key::agrave - key::Agrave,
361        key::Ooblique..=key::Thorn => lower += key::oslash - key::Ooblique,
362        key::oslash..=key::thorn => upper -= key::oslash - key::Ooblique,
363        key::Aogonek => lower = key::aogonek,
364        key::aogonek => upper = key::Aogonek,
365        key::Lstroke..=key::Sacute => lower += key::lstroke - key::Lstroke,
366        key::lstroke..=key::sacute => upper -= key::lstroke - key::Lstroke,
367        key::Scaron..=key::Zacute => lower += key::scaron - key::Scaron,
368        key::scaron..=key::zacute => upper -= key::scaron - key::Scaron,
369        key::Zcaron..=key::Zabovedot => lower += key::zcaron - key::Zcaron,
370        key::zcaron..=key::zabovedot => upper -= key::zcaron - key::Zcaron,
371        key::Racute..=key::Tcedilla => lower += key::racute - key::Racute,
372        key::racute..=key::tcedilla => upper -= key::racute - key::Racute,
373        key::Hstroke..=key::Hcircumflex => lower += key::hstroke - key::Hstroke,
374        key::hstroke..=key::hcircumflex => upper -= key::hstroke - key::Hstroke,
375        key::Gbreve..=key::Jcircumflex => lower += key::gbreve - key::Gbreve,
376        key::gbreve..=key::jcircumflex => upper -= key::gbreve - key::Gbreve,
377        key::Cabovedot..=key::Scircumflex => lower += key::cabovedot - key::Cabovedot,
378        key::cabovedot..=key::scircumflex => upper -= key::cabovedot - key::Cabovedot,
379        key::Rcedilla..=key::Tslash => lower += key::rcedilla - key::Rcedilla,
380        key::rcedilla..=key::tslash => upper -= key::rcedilla - key::Rcedilla,
381        key::ENG => lower = key::eng,
382        key::eng => upper = key::ENG,
383        key::Amacron..=key::Umacron => lower += key::amacron - key::Amacron,
384        key::amacron..=key::umacron => upper -= key::amacron - key::Amacron,
385        key::Serbian_DJE..=key::Serbian_DZE => lower -= key::Serbian_DJE - key::Serbian_dje,
386        key::Serbian_dje..=key::Serbian_dze => upper += key::Serbian_DJE - key::Serbian_dje,
387        key::Cyrillic_YU..=key::Cyrillic_HARDSIGN => lower -= key::Cyrillic_YU - key::Cyrillic_yu,
388        key::Cyrillic_yu..=key::Cyrillic_hardsign => upper += key::Cyrillic_YU - key::Cyrillic_yu,
389        key::Greek_ALPHAaccent..=key::Greek_OMEGAaccent => {
390            lower += key::Greek_alphaaccent - key::Greek_ALPHAaccent
391        }
392        key::Greek_alphaaccent..=key::Greek_omegaaccent
393            if !matches!(
394                keysym.0,
395                key::Greek_iotaaccentdieresis | key::Greek_upsilonaccentdieresis
396            ) =>
397        {
398            upper -= key::Greek_alphaaccent - key::Greek_ALPHAaccent
399        }
400        key::Greek_ALPHA..=key::Greek_OMEGA => lower += key::Greek_alpha - key::Greek_ALPHA,
401        key::Greek_alpha..=key::Greek_omega if !matches!(keysym.0, key::Greek_finalsmallsigma) => {
402            upper -= key::Greek_alpha - key::Greek_ALPHA
403        }
404        key::Armenian_AYB..=key::Armenian_fe => {
405            lower |= 1;
406            upper &= !1;
407        }
408        _ => {}
409    }
410
411    (Keysym(upper), Keysym(lower))
412}
