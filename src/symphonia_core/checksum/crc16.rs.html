288/// `Crc16Ansi` implements the CRC-16 algorithm using the ANSI polynominal.
289///
290/// * Polynomial = 0x8005
291/// * RefIn = false
292/// * RefOut = false
293/// * XorOut = false
294pub struct Crc16Ansi {
295    state: u16,
296}
297
298impl Crc16Ansi {
299    /// Instantiate a `Crc16Ansi` instance with an initial state.
300    pub fn new(state: u16) -> Self {
301        Crc16Ansi { state }
302    }
303
304    /// Returns the computed CRC.
305    pub fn crc(&self) -> u16 {
306        self.state
307    }
308}
309
310impl Monitor for Crc16Ansi {
311    #[inline(always)]
312    fn process_byte(&mut self, byte: u8) {
313        self.state = (self.state << 8) ^ CRC16_ANSI[0][((self.state >> 8) as u8 ^ byte) as usize];
314    }
315
316    fn process_buf_bytes(&mut self, buf: &[u8]) {
317        let mut iter = buf.chunks_exact(8);
318
319        for bytes in &mut iter {
320            let word = (u16::from(bytes[0]) << 8) | u16::from(bytes[1]);
321
322            let xor = self.state ^ word;
323
324            self.state = CRC16_ANSI[7][((xor >> 8) & 0xff) as usize]
325                ^ CRC16_ANSI[6][((xor >> 0) & 0xff) as usize]
326                ^ CRC16_ANSI[5][usize::from(bytes[2])]
327                ^ CRC16_ANSI[4][usize::from(bytes[3])]
328                ^ CRC16_ANSI[3][usize::from(bytes[4])]
329                ^ CRC16_ANSI[2][usize::from(bytes[5])]
330                ^ CRC16_ANSI[1][usize::from(bytes[6])]
331                ^ CRC16_ANSI[0][usize::from(bytes[7])];
332        }
333
334        for byte in iter.remainder() {
335            self.process_byte(*byte);
336        }
337    }
338}
