1//! Find "simple" numbers is some range. Used by sliders.
2
3const NUM_DECIMALS: usize = 15;
4
5/// Find the "simplest" number in a closed range [min, max], i.e. the one with the fewest decimal digits.
6///
7/// So in the range `[0.83, 1.354]` you will get `1.0`, and for `[0.37, 0.48]` you will get `0.4`.
8/// This is used when dragging sliders etc to get the values that users are most likely to desire.
9/// This assumes a decimal centric user.
10pub fn best_in_range_f64(min: f64, max: f64) -> f64 {
11    // Avoid NaN if we can:
12    if min.is_nan() {
13        return max;
14    }
15    if max.is_nan() {
16        return min;
17    }
18
19    if max < min {
20        return best_in_range_f64(max, min);
21    }
22    if min == max {
23        return min;
24    }
25    if min <= 0.0 && 0.0 <= max {
26        return 0.0; // always prefer zero
27    }
28    if min < 0.0 {
29        return -best_in_range_f64(-max, -min);
30    }
31
32    // Prefer finite numbers:
33    if !max.is_finite() {
34        return min;
35    }
36    debug_assert!(min.is_finite() && max.is_finite());
37
38    let min_exponent = min.log10();
39    let max_exponent = max.log10();
40
41    if min_exponent.floor() != max_exponent.floor() {
42        // pick the geometric center of the two:
43        let exponent = (min_exponent + max_exponent) / 2.0;
44        return 10.0_f64.powi(exponent.round() as i32);
45    }
46
47    if is_integer(min_exponent) {
48        return 10.0_f64.powf(min_exponent);
49    }
50    if is_integer(max_exponent) {
51        return 10.0_f64.powf(max_exponent);
52    }
53
54    let exp_factor = 10.0_f64.powi(max_exponent.floor() as i32);
55
56    let min_str = to_decimal_string(min / exp_factor);
57    let max_str = to_decimal_string(max / exp_factor);
58
59    let mut ret_str = [0; NUM_DECIMALS];
60
61    // Select the common prefix:
62    let mut i = 0;
63    while i < NUM_DECIMALS && max_str[i] == min_str[i] {
64        ret_str[i] = max_str[i];
65        i += 1;
66    }
67
68    if i < NUM_DECIMALS {
69        // Pick the deciding digit.
70        // Note that "to_decimal_string" rounds down, so we that's why we add 1 here
71        ret_str[i] = simplest_digit_closed_range(min_str[i] + 1, max_str[i]);
72    }
73
74    from_decimal_string(&ret_str) * exp_factor
75}
76
77fn is_integer(f: f64) -> bool {
78    f.round() == f
79}
80
81fn to_decimal_string(v: f64) -> [i32; NUM_DECIMALS] {
82    debug_assert!(v < 10.0, "{v:?}");
83    let mut digits = [0; NUM_DECIMALS];
84    let mut v = v.abs();
85    for r in &mut digits {
86        let digit = v.floor();
87        *r = digit as i32;
88        v -= digit;
89        v *= 10.0;
90    }
91    digits
92}
93
94fn from_decimal_string(s: &[i32]) -> f64 {
95    let mut ret: f64 = 0.0;
96    for (i, &digit) in s.iter().enumerate() {
97        ret += (digit as f64) * 10.0_f64.powi(-(i as i32));
98    }
99    ret
100}

102/// Find the simplest integer in the range [min, max]
103fn simplest_digit_closed_range(min: i32, max: i32) -> i32 {
104    debug_assert!(1 <= min && min <= max && max <= 9);
105    if min <= 5 && 5 <= max {
106        5
107    } else {
108        (min + max) / 2
109    }
110}

112#[allow(clippy::approx_constant)]
113#[test]
114fn test_aim() {
115    assert_eq!(best_in_range_f64(-0.2, 0.0), 0.0, "Prefer zero");
116    assert_eq!(best_in_range_f64(-10_004.23, 3.14), 0.0, "Prefer zero");
117    assert_eq!(best_in_range_f64(-0.2, 100.0), 0.0, "Prefer zero");
118    assert_eq!(best_in_range_f64(0.2, 0.0), 0.0, "Prefer zero");
119    assert_eq!(best_in_range_f64(7.8, 17.8), 10.0);
120    assert_eq!(best_in_range_f64(99.0, 300.0), 100.0);
121    assert_eq!(best_in_range_f64(-99.0, -300.0), -100.0);
122    assert_eq!(best_in_range_f64(0.4, 0.9), 0.5, "Prefer ending on 5");
123    assert_eq!(best_in_range_f64(14.1, 19.99), 15.0, "Prefer ending on 5");
124    assert_eq!(best_in_range_f64(12.3, 65.9), 50.0, "Prefer leading 5");
125    assert_eq!(best_in_range_f64(493.0, 879.0), 500.0, "Prefer leading 5");
126    assert_eq!(best_in_range_f64(0.37, 0.48), 0.40);
127    // assert_eq!(best_in_range_f64(123.71, 123.76), 123.75); // TODO(emilk): we get 123.74999999999999 here
128    // assert_eq!(best_in_range_f32(123.71, 123.76), 123.75);
129    assert_eq!(best_in_range_f64(7.5, 16.3), 10.0);
130    assert_eq!(best_in_range_f64(7.5, 76.3), 10.0);
131    assert_eq!(best_in_range_f64(7.5, 763.3), 100.0);
132    assert_eq!(best_in_range_f64(7.5, 1_345.0), 100.0);
133    assert_eq!(best_in_range_f64(7.5, 123_456.0), 1000.0, "Geometric mean");
134    assert_eq!(best_in_range_f64(9.9999, 99.999), 10.0);
135    assert_eq!(best_in_range_f64(10.000, 99.999), 10.0);
136    assert_eq!(best_in_range_f64(10.001, 99.999), 50.0);
137    assert_eq!(best_in_range_f64(10.001, 100.000), 100.0);
138    assert_eq!(best_in_range_f64(99.999, 100.000), 100.0);
139    assert_eq!(best_in_range_f64(10.001, 100.001), 100.0);
140
141    const NAN: f64 = f64::NAN;
142    const INFINITY: f64 = f64::INFINITY;
143    const NEG_INFINITY: f64 = f64::NEG_INFINITY;
144    assert!(best_in_range_f64(NAN, NAN).is_nan());
145    assert_eq!(best_in_range_f64(NAN, 1.2), 1.2);
146    assert_eq!(best_in_range_f64(NAN, INFINITY), INFINITY);
147    assert_eq!(best_in_range_f64(1.2, NAN), 1.2);
148    assert_eq!(best_in_range_f64(1.2, INFINITY), 1.2);
149    assert_eq!(best_in_range_f64(INFINITY, 1.2), 1.2);
150    assert_eq!(best_in_range_f64(NEG_INFINITY, 1.2), 0.0);
151    assert_eq!(best_in_range_f64(NEG_INFINITY, -2.7), -2.7);
152    assert_eq!(best_in_range_f64(INFINITY, INFINITY), INFINITY);
153    assert_eq!(best_in_range_f64(NEG_INFINITY, NEG_INFINITY), NEG_INFINITY);
154    assert_eq!(best_in_range_f64(NEG_INFINITY, INFINITY), 0.0);
155    assert_eq!(best_in_range_f64(INFINITY, NEG_INFINITY), 0.0);
156}
