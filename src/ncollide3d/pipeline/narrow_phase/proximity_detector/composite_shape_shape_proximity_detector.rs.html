10/// Proximity detector between a concave shape and another shape.
11pub struct CompositeShapeShapeProximityDetector<N> {
12    sub_detectors: HashMap<usize, ProximityAlgorithm<N>, DeterministicState>,
13    to_delete: Vec<usize>,
14    interferences: Vec<usize>,
15    intersecting_key: usize,
16    flip: bool,
17}
18
19impl<N> CompositeShapeShapeProximityDetector<N> {
20    /// Creates a new proximity detector between a concave shape and another shape.
21    pub fn new(flip: bool) -> CompositeShapeShapeProximityDetector<N> {
22        CompositeShapeShapeProximityDetector {
23            sub_detectors: HashMap::with_hasher(DeterministicState),
24            to_delete: Vec::new(),
25            interferences: Vec::new(),
26            intersecting_key: usize::max_value(),
27            flip,
28        }
29    }
30}

32impl<N: RealField + Copy> CompositeShapeShapeProximityDetector<N> {
33    fn do_update(
34        &mut self,
35        dispatcher: &dyn ProximityDispatcher<N>,
36        m1: &Isometry<N>,
37        g1: &dyn CompositeShape<N>,
38        m2: &Isometry<N>,
39        g2: &dyn Shape<N>,
40        margin: N,
41        flip: bool,
42    ) -> Option<Proximity> {
43        let mut result = Proximity::Disjoint;
44
45        // Remove outdated sub detectors.
46        for key in self.to_delete.iter() {
47            let _ = self.sub_detectors.remove(key);
48        }
49
50        self.to_delete.clear();
51        self.interferences.clear();
52
53        // First, test if the previously intersecting shapes are still intersecting.
54        if self.intersecting_key != usize::max_value() {
55            let detector = self.sub_detectors.get_mut(&self.intersecting_key).unwrap();
56            let mut prox = None;
57            g1.map_part_at(self.intersecting_key, m1, &mut |m1, g1| {
58                prox = detector.update(dispatcher, m1, g1, m2, g2, margin)
59            });
60
61            match prox? {
62                Proximity::Intersecting => return Some(Proximity::Intersecting), // Early return.
63                Proximity::WithinMargin => result = Proximity::WithinMargin,
64                Proximity::Disjoint => {}
65            }
66        }
67
68        let m12 = m1.inverse() * m2.clone();
69        let ls_aabb2 = bounding_volume::aabb(g2, &m12).loosened(margin);
70
71        // Update all collisions
72        for detector in &mut self.sub_detectors {
73            let key = *detector.0;
74
75            if key == self.intersecting_key {
76                // We already dealt with that one.
77                continue;
78            }
79
80            if ls_aabb2.intersects(&g1.aabb_at(key)) {
81                let mut prox = None;
82
83                g1.map_part_at(key, m1, &mut |m1, g1| {
84                    prox = detector.1.update(dispatcher, m1, g1, m2, g2, margin)
85                });
86
87                match prox? {
88                    Proximity::Intersecting => {
89                        self.intersecting_key = *detector.0;
90                        return Some(Proximity::Intersecting); // No need to search any further.
91                    }
92                    Proximity::WithinMargin => result = Proximity::WithinMargin,
93                    Proximity::Disjoint => {}
94                }
95            } else {
96                // FIXME: ask the detector if it wants to be removed or not
97                self.to_delete.push(key);
98            }
99        }
100
101        // Find new proximities.
102        {
103            let mut visitor =
104                BoundingVolumeInterferencesCollector::new(&ls_aabb2, &mut self.interferences);
105            g1.bvh().visit(&mut visitor);
106        }
107
108        for key in &self.interferences {
109            let entry = self.sub_detectors.entry(*key);
110            let detector = match entry {
111                Entry::Occupied(entry) => Some(entry.into_mut()),
112                Entry::Vacant(entry) => {
113                    let mut new_detector = None;
114
115                    g1.map_part_at(*key, &Isometry::identity(), &mut |_, g1| {
116                        if flip {
117                            new_detector = dispatcher.get_proximity_algorithm(g2, g1)
118                        } else {
119                            new_detector = dispatcher.get_proximity_algorithm(g1, g2)
120                        }
121                    });
122
123                    if let Some(new_detector) = new_detector {
124                        Some(entry.insert(new_detector))
125                    } else {
126                        None
127                    }
128                }
129            };
130
131            if let Some(sub_detector) = detector {
132                let mut prox = None;
133
134                g1.map_part_at(*key, m1, &mut |m1, g1| {
135                    if flip {
136                        prox = sub_detector.update(dispatcher, m2, g2, m1, g1, margin);
137                    } else {
138                        prox = sub_detector.update(dispatcher, m1, g1, m2, g2, margin);
139                    }
140                });
141
142                match prox? {
143                    Proximity::Intersecting => {
144                        self.intersecting_key = *key;
145                        return Some(Proximity::Intersecting); // No need to search further.
146                    }
147                    Proximity::WithinMargin => result = Proximity::WithinMargin,
148                    Proximity::Disjoint => {}
149                }
150            }
151        }
152
153        // Disjoints or within margin.
154        self.intersecting_key = usize::max_value();
155        Some(result)
156    }
157}

159impl<N: RealField + Copy> ProximityDetector<N> for CompositeShapeShapeProximityDetector<N> {
160    fn update(
161        &mut self,
162        dispatcher: &dyn ProximityDispatcher<N>,
163        m1: &Isometry<N>,
164        g1: &dyn Shape<N>,
165        m2: &Isometry<N>,
166        g2: &dyn Shape<N>,
167        margin: N,
168    ) -> Option<Proximity> {
169        if !self.flip {
170            let cs = g1.as_composite_shape()?;
171            self.do_update(dispatcher, m1, cs, m2, g2, margin, false)
172        } else {
173            let cs = g2.as_composite_shape()?;
174            self.do_update(dispatcher, m2, cs, m1, g1, margin, true)
175        }
176    }
177}
