10/// Collision detector between g1 plane and g1 shape implementing the `SupportMap` trait.
11#[derive(Clone)]
12pub struct PlaneConvexPolyhedronManifoldGenerator<N: RealField + Copy> {
13    flip: bool,
14    feature: ConvexPolygonalFeature<N>,
15}
16
17impl<N: RealField + Copy> PlaneConvexPolyhedronManifoldGenerator<N> {
18    /// Creates g1 new persistent collision detector between g1 plane and g1 shape with g1 support
19    /// mapping function.
20    #[inline]
21    pub fn new(flip: bool) -> PlaneConvexPolyhedronManifoldGenerator<N> {
22        PlaneConvexPolyhedronManifoldGenerator {
23            flip,
24            feature: ConvexPolygonalFeature::new(),
25        }
26    }
27
28    #[inline]
29    fn do_update_to(
30        m1: &Isometry<N>,
31        g1: &dyn Shape<N>,
32        proc1: Option<&dyn ContactPreprocessor<N>>,
33        m2: &Isometry<N>,
34        g2: &dyn Shape<N>,
35        proc2: Option<&dyn ContactPreprocessor<N>>,
36        prediction: &ContactPrediction<N>,
37        poly_feature: &mut ConvexPolygonalFeature<N>,
38        manifold: &mut ContactManifold<N>,
39        flip: bool,
40    ) -> bool {
41        if let (Some(plane), Some(cp)) = (g1.as_shape::<Plane<N>>(), g2.as_convex_polyhedron()) {
42            let plane_normal = m1 * plane.normal;
43            let plane_center = Point::from(m1.translation.vector);
44
45            cp.support_face_toward(m2, &-plane_normal, poly_feature);
46
47            for (i, world2) in poly_feature.vertices.iter().enumerate() {
48                let dpt = *world2 - plane_center;
49                let dist = dpt.dot(plane_normal.as_ref());
50
51                if dist <= prediction.linear() {
52                    let world1 = *world2 + (-*plane_normal * dist);
53                    let local1 = m1.inverse_transform_point(&world1);
54                    let local2 = m2.inverse_transform_point(&world2);
55                    let f1 = FeatureId::Face(0);
56                    let f2 = poly_feature.vertices_id[i];
57                    let mut kinematic = ContactKinematic::new();
58                    let contact;
59
60                    let approx_plane = NeighborhoodGeometry::Plane(plane.normal);
61                    let approx2 = NeighborhoodGeometry::Point;
62
63                    if !flip {
64                        contact = Contact::new(world1, *world2, plane_normal, -dist);
65                        kinematic.set_approx1(f1, local1, approx_plane);
66                        kinematic.set_approx2(f2, local2, approx2);
67                        let _ = manifold.push(contact, kinematic, local2, proc1, proc2);
68                    } else {
69                        contact = Contact::new(*world2, world1, -plane_normal, -dist);
70                        kinematic.set_approx1(f2, local2, approx2);
71                        kinematic.set_approx2(f1, local1, approx_plane);
72                        let _ = manifold.push(contact, kinematic, local2, proc2, proc1);
73                    }
74                }
75            }
76
77            true
78        } else {
79            false
80        }
81    }
82}

84impl<N: RealField + Copy> ContactManifoldGenerator<N>
85    for PlaneConvexPolyhedronManifoldGenerator<N>
86{
87    fn generate_contacts(
88        &mut self,
89        _: &dyn ContactDispatcher<N>,
90        m1: &Isometry<N>,
91        g1: &dyn Shape<N>,
92        proc1: Option<&dyn ContactPreprocessor<N>>,
93        m2: &Isometry<N>,
94        g2: &dyn Shape<N>,
95        proc2: Option<&dyn ContactPreprocessor<N>>,
96        prediction: &ContactPrediction<N>,
97        manifold: &mut ContactManifold<N>,
98    ) -> bool {
99        if !self.flip {
100            Self::do_update_to(
101                m1,
102                g1,
103                proc1,
104                m2,
105                g2,
106                proc2,
107                prediction,
108                &mut self.feature,
109                manifold,
110                false,
111            )
112        } else {
113            Self::do_update_to(
114                m2,
115                g2,
116                proc2,
117                m1,
118                g1,
119                proc1,
120                prediction,
121                &mut self.feature,
122                manifold,
123                true,
124            )
125        }
126    }
