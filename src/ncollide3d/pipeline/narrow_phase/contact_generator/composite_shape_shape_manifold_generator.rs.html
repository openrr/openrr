15/// Collision detector between a concave shape and another shape.
16pub struct CompositeShapeShapeManifoldGenerator<N: RealField + Copy> {
17    sub_detectors: HashMap<usize, (ContactAlgorithm<N>, usize), DeterministicState>,
18    interferences: Vec<usize>,
19    flip: bool,
20    timestamp: usize,
21}
22
23impl<N: RealField + Copy> CompositeShapeShapeManifoldGenerator<N> {
24    /// Creates a new collision detector between a concave shape and another shape.
25    pub fn new(flip: bool) -> CompositeShapeShapeManifoldGenerator<N> {
26        CompositeShapeShapeManifoldGenerator {
27            sub_detectors: HashMap::with_hasher(DeterministicState),
28            interferences: Vec::new(),
29            flip,
30            timestamp: 0,
31        }
32    }
33
34    fn do_update(
35        &mut self,
36        dispatcher: &dyn ContactDispatcher<N>,
37        m1: &Isometry<N>,
38        g1: &dyn CompositeShape<N>,
39        proc1: Option<&dyn ContactPreprocessor<N>>,
40        m2: &Isometry<N>,
41        g2: &dyn Shape<N>,
42        proc2: Option<&dyn ContactPreprocessor<N>>,
43        prediction: &ContactPrediction<N>,
44        manifold: &mut ContactManifold<N>,
45        flip: bool,
46    ) {
47        self.timestamp += 1;
48
49        // Find new collisions
50        let ls_m2 = m1.inverse() * m2.clone();
51        let ls_aabb2 = bounding_volume::aabb(g2, &ls_m2).loosened(prediction.linear());
52
53        {
54            let mut visitor =
55                BoundingVolumeInterferencesCollector::new(&ls_aabb2, &mut self.interferences);
56            g1.bvh().visit(&mut visitor);
57        }
58
59        for i in self.interferences.drain(..) {
60            match self.sub_detectors.entry(i) {
61                Entry::Occupied(mut entry) => entry.get_mut().1 = self.timestamp,
62                Entry::Vacant(entry) => {
63                    let mut new_detector = None;
64
65                    g1.map_part_at(i, &Isometry::identity(), &mut |_, g1| {
66                        if flip {
67                            new_detector = dispatcher.get_contact_algorithm(g2, g1)
68                        } else {
69                            new_detector = dispatcher.get_contact_algorithm(g1, g2)
70                        }
71                    });
72
73                    if let Some(new_detector) = new_detector {
74                        let _ = entry.insert((new_detector, self.timestamp));
75                    }
76                }
77            }
78        }
79
80        // Update all collisions
81        let timestamp = self.timestamp;
82
83        self.sub_detectors.retain(|key, detector| {
84            if detector.1 != timestamp {
85                // FIXME: ask the detector if it wants to be removed or not
86                false
87            } else {
88                let mut keep = false;
89                g1.map_part_and_preprocessor_at(*key, m1, prediction, &mut |m1, g1, part_proc1| {
90                    keep = if flip {
91                        detector.0.generate_contacts(
92                            dispatcher,
93                            m2,
94                            g2,
95                            proc2,
96                            m1,
97                            g1,
98                            Some(&(proc1, part_proc1)),
99                            prediction,
100                            manifold,
101                        )
102                    } else {
103                        detector.0.generate_contacts(
104                            dispatcher,
105                            m1,
106                            g1,
107                            Some(&(proc1, part_proc1)),
108                            m2,
109                            g2,
110                            proc2,
111                            prediction,
112                            manifold,
113                        )
114                    }
115                });
116
117                keep
118            }
119        });
120    }
121}

123impl<N: RealField + Copy> ContactManifoldGenerator<N> for CompositeShapeShapeManifoldGenerator<N> {
124    fn generate_contacts(
125        &mut self,
126        d: &dyn ContactDispatcher<N>,
127        ma: &Isometry<N>,
128        a: &dyn Shape<N>,
129        proc1: Option<&dyn ContactPreprocessor<N>>,
130        mb: &Isometry<N>,
131        b: &dyn Shape<N>,
132        proc2: Option<&dyn ContactPreprocessor<N>>,
133        prediction: &ContactPrediction<N>,
134        manifold: &mut ContactManifold<N>,
135    ) -> bool {
136        if !self.flip {
137            if let Some(cs) = a.as_composite_shape() {
138                self.do_update(d, ma, cs, proc1, mb, b, proc2, prediction, manifold, false);
139                return true;
140            }
141        } else {
142            if let Some(cs) = b.as_composite_shape() {
143                self.do_update(d, mb, cs, proc2, ma, a, proc1, prediction, manifold, true);
144                return true;
145            }
146        }
147
148        return false;
149    }
150
151    fn init_manifold(&self) -> ContactManifold<N> {
152        let mut res = ContactManifold::new();
153        res.set_tracking_mode(ContactTrackingMode::FeatureBased);
154        res
155    }
