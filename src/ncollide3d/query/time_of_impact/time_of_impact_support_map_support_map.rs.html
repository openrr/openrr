1use na::{RealField, Unit};
2
3use crate::math::{Isometry, Vector};
4use crate::query::algorithms::gjk::{self, GJKResult};
5use crate::query::algorithms::{special_support_maps::DilatedShape, VoronoiSimplex};
6use crate::query::{self, TOIStatus, TOI};
7use crate::shape::SupportMap;
8
9/// Time of impacts between two support-mapped shapes under translational movement.
10pub fn time_of_impact_support_map_support_map<N, G1: ?Sized, G2: ?Sized>(
11    m1: &Isometry<N>,
12    vel1: &Vector<N>,
13    g1: &G1,
14    m2: &Isometry<N>,
15    vel2: &Vector<N>,
16    g2: &G2,
17    max_toi: N,
18    target_distance: N,
19) -> Option<TOI<N>>
20where
21    N: RealField + Copy,
22    G1: SupportMap<N>,
23    G2: SupportMap<N>,
24{
25    let dvel = vel2 - vel1;
26
27    if target_distance.is_zero() {
28        gjk::directional_distance(m1, g1, m2, g2, &dvel, &mut VoronoiSimplex::new()).and_then(
29            |(toi, normal, witness1, witness2)| {
30                if toi > max_toi {
31                    None
32                } else {
33                    Some(TOI {
34                        toi,
35                        normal1: Unit::new_unchecked(m1.inverse_transform_vector(&normal)),
36                        normal2: Unit::new_unchecked(m2.inverse_transform_vector(&-normal)),
37                        witness1: m1.inverse_transform_point(&witness1),
38                        witness2: m2.inverse_transform_point(&witness2),
39                        status: if toi.is_zero() {
40                            TOIStatus::Penetrating
41                        } else {
42                            TOIStatus::Converged
43                        }, // XXX
44                    })
45                }
46            },
47        )
48    } else {
49        let dilated1 = DilatedShape {
50            shape: g1,
51            radius: target_distance,
52        };
53
54        gjk::directional_distance(m1, &dilated1, m2, g2, &dvel, &mut VoronoiSimplex::new())
55            .and_then(|(toi, normal, witness1, witness2)| {
56                if toi > max_toi {
57                    None
58                } else {
59                    // This is mutable because, if the TOI is zero, we have to determine
60                    // if the status isn't really a TOIStatus::Penetrating.
61                    let mut status = TOIStatus::Converged;
62
63                    if toi.is_zero() {
64                        // The TOI is zero but we don't have valid witness points and normal
65                        // yet because of we based our computations so far on the dilated shape.
66                        // Therefore we need an extra step to retrieve the actual closest points, and
67                        // determine if the actual shapes are penetrating or not.
68                        // FIXME: all those computations are costly. Add a variant that returns only
69                        // the TOI and does not computes the normal and witness points?
70                        match query::closest_points_support_map_support_map_with_params(
71                            m1,
72                            g1,
73                            m2,
74                            g2,
75                            target_distance,
76                            &mut VoronoiSimplex::new(),
77                            Some(normal),
78                        ) {
79                            GJKResult::ClosestPoints(pt1, pt2, _) => {
80                                // Ok, we managed to compute the witness points.
81                                let normal = Unit::new_normalize(pt2 - pt1);
82                                let normal1 =
83                                    Unit::new_unchecked(m1.inverse_transform_vector(&normal));
84                                return Some(TOI {
85                                    toi,
86                                    normal1,
87                                    normal2: Unit::new_unchecked(
88                                        m2.inverse_transform_vector(&-normal),
89                                    ),
90                                    witness1: m1.inverse_transform_point(&witness1),
91                                    witness2: m2.inverse_transform_point(&witness2),
92                                    status: TOIStatus::Converged,
93                                });
94                            }
95                            GJKResult::NoIntersection(_) => {
96                                // This should never happen.
97                            }
98                            GJKResult::Intersection => status = TOIStatus::Penetrating,
99                            GJKResult::Proximity(_) => unreachable!(),
100                        }
101                    }
102
103                    let normal1 = Unit::new_unchecked(m1.inverse_transform_vector(&normal));
104                    Some(TOI {
105                        toi,
106                        normal1,
107                        normal2: Unit::new_unchecked(m2.inverse_transform_vector(&-normal)),
108                        witness1: m1.inverse_transform_point(&witness1)
109                            - *normal1 * target_distance,
110                        witness2: m2.inverse_transform_point(&witness2),
111                        status,
112                    })
113                }
114            })
115    }
116}
