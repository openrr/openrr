33/// Time of impacts between two support-mapped shapes under a rigid motion.
34///
35/// You probably want to use `query::nonlinear_time_of_impact_support_map_support_map` instead of this one.
36/// The distance function between the two shapes must be given.
37pub fn nonlinear_time_of_impact_support_map_support_map_with_closest_points_function<
38    N,
39    G1: ?Sized,
40    G2: ?Sized,
41>(
42    motion1: &(impl RigidMotion<N> + ?Sized),
43    g1: &G1,
44    motion2: &(impl RigidMotion<N> + ?Sized),
45    g2: &G2,
46    max_toi: N,
47    target_distance: N,
48    closest_points: impl Fn(&Isometry<N>, &G1, &Isometry<N>, &G2, N) -> ClosestPoints<N>,
49) -> Option<TOI<N>>
50where
51    N: RealField + Copy,
52    G1: SupportMap<N>,
53    G2: SupportMap<N>,
54{
55    let _0_5 = na::convert(0.5);
56    let mut min_t = N::zero();
57    let mut prev_min_t = min_t;
58    let abs_tol: N = query::algorithms::gjk::eps_tol();
59    let rel_tol = abs_tol.sqrt();
60    let mut result = TOI {
61        toi: N::zero(),
62        normal1: Vector::x_axis(),
63        normal2: Vector::x_axis(),
64        witness1: Point::origin(),
65        witness2: Point::origin(),
66        status: TOIStatus::Penetrating,
67    };
68
69    loop {
70        let pos1 = motion1.position_at_time(result.toi);
71        let pos2 = motion2.position_at_time(result.toi);
72
73        // FIXME: use the _with_params version of the closest points query.
74        match closest_points(&pos1, g1, &pos2, g2, N::max_value().unwrap()) {
75            ClosestPoints::Intersecting => {
76                if result.toi == N::zero() {
77                    result.status = TOIStatus::Penetrating
78                } else {
79                    result.status = TOIStatus::Failed;
80                }
81                break;
82            }
83            ClosestPoints::WithinMargin(p1, p2) => {
84                // FIXME: do the "inverse_transform_point" only when we are about to return.
85                // the result.
86                result.witness1 = pos1.inverse_transform_point(&p1);
87                result.witness2 = pos2.inverse_transform_point(&p2);
88
89                if let Some((dir, mut dist)) = Unit::try_new_and_get(p2 - p1, N::default_epsilon())
90                {
91                    // FIXME: do the "inverse transform unit vector" only when we are about to return.
92                    result.normal1 = pos1.inverse_transform_unit_vector(&dir);
93                    result.normal2 = pos2.inverse_transform_unit_vector(&-dir);
94
95                    let mut niter = 0;
96                    min_t = result.toi;
97                    let mut max_t = max_toi;
98                    let min_target_distance = (target_distance - rel_tol).max(N::zero());
99                    let max_target_distance = target_distance + rel_tol;
100
101                    loop {
102                        // FIXME: use the secant method too for finding the next iterate.
103                        if dist < min_target_distance {
104                            // Too close or penetration, go back in time.
105                            max_t = result.toi;
106                            result.toi = (min_t + result.toi) * _0_5;
107                        } else if dist > max_target_distance {
108                            // Too far apart, go forward in time.
109                            min_t = result.toi;
110                            result.toi = (result.toi + max_t) * _0_5;
111                        } else {
112                            // Reached tolerance, break.
113                            break;
114                        }
115
116                        if max_t - min_t < abs_tol {
117                            result.toi = min_t;
118                            break;
119                        }
120
121                        let pos1 = motion1.position_at_time(result.toi);
122                        let pos2 = motion2.position_at_time(result.toi);
123                        let pt1 = g1.support_point_toward(&pos1, &dir);
124                        let pt2 = g2.support_point_toward(&pos2, &-dir);
125
126                        dist = (pt2 - pt1).dot(&dir);
127
128                        niter += 1;
129                    }
130
131                    if min_t - prev_min_t < abs_tol {
132                        if max_t == max_toi {
133                            let pos1 = motion1.position_at_time(max_t);
134                            let pos2 = motion2.position_at_time(max_t);
135                            // Check the configuration at max_t to see if the object are not disjoint.
136                            // NOTE: could we do this earlier, before the above loop?
137                            // It feels like this could prevent catching some corner-cases like
138                            // if one object is rotated by almost 180 degrees while the other is immobile.
139                            let pt1 = g1.support_point_toward(&pos1, &dir);
140                            let pt2 = g2.support_point_toward(&pos2, &-dir);
141                            if (pt2 - pt1).dot(&dir) > target_distance {
142                                // We found an axis that separate both objects at the end configuration.
143                                return None;
144                            }
145                        }
146
147                        result.status = TOIStatus::Converged;
148                        break;
149                    }
150
151                    prev_min_t = min_t;
152
153                    if niter == 0 {
154                        result.status = TOIStatus::Converged;
155                        break;
156                    }
157                } else {
158                    result.status = TOIStatus::Failed;
159                    break;
160                }
161            }
162            ClosestPoints::Disjoint => unreachable!(),
163        }
164    }
165
166    Some(result)
167}
