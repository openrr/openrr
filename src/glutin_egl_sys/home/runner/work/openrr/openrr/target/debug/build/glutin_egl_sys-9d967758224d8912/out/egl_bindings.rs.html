311        impl FnPtr {
312            /// Creates a `FnPtr` from a load attempt.
313            fn new(ptr: *const __gl_imports::raw::c_void) -> FnPtr {
314                if ptr.is_null() {
315                    FnPtr {
316                        f: missing_fn_panic as *const __gl_imports::raw::c_void,
317                        is_loaded: false
318                    }
319                } else {
320                    FnPtr { f: ptr, is_loaded: true }
321                }
322            }

325            ///
326            /// If it returns `false`, calling the corresponding function will fail.
327            #[inline]

329            pub fn is_loaded(&self) -> bool {
330                self.is_loaded
331            }
332        }
