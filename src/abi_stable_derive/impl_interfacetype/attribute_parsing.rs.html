7use crate::{
8    impl_interfacetype::{TraitStruct, WhichTrait, TRAIT_LIST},
9    parse_utils::parse_str_as_ident,
10};

12/// The traits that are (un)implemented,parsed from the `#[sabi(impl_InterfaceType())]`
13/// helper attribute.
14pub(crate) struct ImplInterfaceType {
15    pub(crate) impld: Vec<Ident>,
16    pub(crate) unimpld: Vec<Ident>,
17}
18
19/// Parses the `#[sabi(impl_InterfaceType())]` helper attribute.
20pub(crate) fn parse_impl_interfacetype(
21    input: &ParseBuffer<'_>,
22) -> Result<ImplInterfaceType, syn::Error> {
23    let trait_map = TRAIT_LIST
24        .iter()
25        .map(|t| {
26            let ident = parse_str_as_ident(t.name);
27            (ident, t.which_trait)
28        })
29        .collect::<HashMap<Ident, WhichTrait>>();
30
31    let mut impld_struct = TraitStruct::TRAITS.map(|_, _| false);
32
33    let mut impld = Vec::new();
34    let mut unimpld = Vec::new();
35
36    let valid_traits = || -> String {
37        trait_map
38            .keys()
39            .map(|x| x.to_string())
40            .collect::<Vec<String>>()
41            .join("\n    ")
42    };
43
44    input.for_each_separated(Token!(,), |input| {
45        let trait_ident = input.parse::<Ident>().map_err(|e| {
46            syn_err!(
47                e.span(),
48                "invalid attribute inside #[sabi(impl_InterfaceType(  ))].\n\
49                     Valid traits:\n    {}\n\
50                    ",
51                valid_traits()
52            )
53        })?;
54
55        match trait_map.get(&trait_ident) {
56            Some(&which_trait) => {
57                impld_struct[which_trait] = true;
58
59                match which_trait {
60                    WhichTrait::Iterator | WhichTrait::DoubleEndedIterator => {
61                        impld_struct.iterator = true;
62                    }
63                    WhichTrait::Eq | WhichTrait::PartialOrd => {
64                        impld_struct.partial_eq = true;
65                    }
66                    WhichTrait::Ord => {
67                        impld_struct.partial_eq = true;
68                        impld_struct.eq = true;
69                        impld_struct.partial_ord = true;
70                    }
71                    WhichTrait::IoBufRead => {
72                        impld_struct.io_read = true;
73                    }
74                    WhichTrait::Error => {
75                        impld_struct.display = true;
76                        impld_struct.debug = true;
77                    }
78                    _ => {}
79                }
80            }
81            None => return_syn_err!(
82                trait_ident.span(),
83                "invalid trait inside #[sabi(impl_InterfaceType(  ))]:\n\
84                 Valid traits:\n    {}\n",
85                valid_traits(),
86            ),
87        }
88        Ok(())
89    })?;
90
91    for (trait_, which_trait) in trait_map {
92        if impld_struct[which_trait] {
93            &mut impld
94        } else {
95            &mut unimpld
96        }
97        .push(trait_);
98    }
99
100    Ok(ImplInterfaceType { impld, unimpld })
101}
