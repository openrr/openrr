29/// Configuration for code generation related to prefix-types.
30pub(crate) struct PrefixKind<'a> {
31    pub(crate) first_suffix_field: FirstSuffixField,
32    pub(crate) prefix_ref: &'a Ident,
33    pub(crate) prefix_fields_struct: &'a Ident,
34    pub(crate) replacing_prefix_ref_docs: &'a [syn::Expr],
35    pub(crate) prefix_bounds: Vec<WherePredicate>,
36    pub(crate) fields: FieldMap<AccessorOrMaybe<'a>>,
37    pub(crate) accessor_bounds: FieldMap<Vec<TypeParamBound>>,
38    pub(crate) cond_field_indices: Vec<usize>,
39    pub(crate) enable_field_if: Vec<&'a syn::Expr>,
40    pub(crate) unconditional_bit_mask: u64,
41    pub(crate) prefix_field_conditionality_mask: u64,
42}

44pub(crate) struct PrefixKindCtor<'a> {
45    pub(crate) arenas: &'a Arenas,
46    #[allow(dead_code)]
47    pub(crate) struct_name: &'a Ident,
48    pub(crate) first_suffix_field: FirstSuffixField,
49    pub(crate) prefix_ref: Option<&'a Ident>,
50    pub(crate) prefix_fields: Option<&'a Ident>,
51    pub(crate) replacing_prefix_ref_docs: &'a [syn::Expr],
52    pub(crate) prefix_bounds: Vec<WherePredicate>,
53    pub(crate) fields: FieldMap<AccessorOrMaybe<'a>>,
54    pub(crate) accessor_bounds: FieldMap<Vec<TypeParamBound>>,
55}

57impl<'a> PrefixKindCtor<'a> {
58    pub fn make(self) -> PrefixKind<'a> {
59        let ctor = self;
60        let mut cond_field_indices = Vec::<usize>::new();
61        let mut enable_field_if = Vec::<&syn::Expr>::new();
62        let mut unconditional_bit_mask = 0u64;
63        let mut conditional_bit_mask = 0u64;
64
65        for (index, field) in ctor.fields.iter() {
66            let field_i = index.pos;
67            match (|| field.to_maybe_accessor()?.accessible_if)() {
68                Some(cond) => {
69                    cond_field_indices.push(field_i);
70                    enable_field_if.push(cond);
71                    conditional_bit_mask |= 1u64 << field_i;
72                }
73                None => {
74                    unconditional_bit_mask |= 1u64 << field_i;
75                }
76            }
77        }
78
79        let prefix_field_conditionality_mask =
80            conditional_bit_mask & low_bit_mask_u64(ctor.first_suffix_field.field_pos as u32);
81
82        PrefixKind {
83            first_suffix_field: ctor.first_suffix_field,
84            prefix_ref: ctor.prefix_ref.unwrap_or_else(|| {
85                let ident = format!("{}_Ref", ctor.struct_name);
86                ctor.arenas.alloc(parse_str_as_ident(&ident))
87            }),
88            prefix_fields_struct: ctor.prefix_fields.unwrap_or_else(|| {
89                let ident = format!("{}_Prefix", ctor.struct_name);
90                ctor.arenas.alloc(parse_str_as_ident(&ident))
91            }),
92            replacing_prefix_ref_docs: ctor.replacing_prefix_ref_docs,
93            prefix_bounds: ctor.prefix_bounds,
94            fields: ctor.fields,
95            accessor_bounds: ctor.accessor_bounds,
96            cond_field_indices,
97            enable_field_if,
98            unconditional_bit_mask,
99            prefix_field_conditionality_mask,
100        }
101    }
102}

111/// The different types of prefix-type accessors.
112#[derive(Debug, Copy, Clone, PartialEq, Eq)]
113pub enum AccessorOrMaybe<'a> {
114    /// Unconditionally returns the field.
115    Accessor,
116    /// Either optionally returns the field,or it does some action when it's missing.
117    Maybe(MaybeAccessor<'a>),
118}
119
120/// Describes a field accessor which is either optional or
121/// does some action when the field is missing.
122#[derive(Debug, Copy, Clone, Default, PartialEq, Eq)]
123pub struct MaybeAccessor<'a> {
124    /// If Some,it uses a bool constant to determine whether a field is accessible.
125    accessible_if: Option<&'a syn::Expr>,
126    /// What the accessor method does when the field is missing.
127    on_missing: OnMissingField<'a>,
128}
129
130#[derive(Copy, Clone, Default, PartialEq, Eq)]
131pub(crate) struct FirstSuffixField {
132    pub(crate) field_pos: usize,
133}
134
135/// What happens in a Prefix-type field getter if the field does not exist.
136#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
137pub(crate) enum OnMissingField<'a> {
138    /// Returns an `Option<FieldType>`,where it returns None if the field is absent.
139    ReturnOption,
140    /// Panics with a default message.
141    Panic,
142    /// Evaluates `function()`,and returns the return value of the function.
143    With { function: &'a syn::Path },
144    /// Returns `some_expression`.
145    Value { value: &'a syn::Expr },
146    /// Returns Default::default
147    Default_,
148}
149
150impl<'a> Default for OnMissingField<'a> {
151    fn default() -> Self {
152        OnMissingField::ReturnOption
153    }
154}
155
156impl<'a> AccessorOrMaybe<'a> {
157    pub(crate) fn new(
158        field_i: FieldIndex,
159        first_suffix_field: FirstSuffixField,
160        pkf: PrefixKindField<'a>,
161        default_omf: OnMissingField<'a>,
162    ) -> Self {
163        if field_i.pos < first_suffix_field.field_pos
164            && pkf.accessible_if.is_none()
165            && pkf.on_missing != Some(OnMissingField::ReturnOption)
166        {
167            AccessorOrMaybe::Accessor
168        } else {
169            AccessorOrMaybe::Maybe(MaybeAccessor {
170                accessible_if: pkf.accessible_if,
171                on_missing: pkf.on_missing.unwrap_or(default_omf),
172            })
173        }
174    }
175
176    #[allow(dead_code)]
177    pub(crate) fn is_conditional(&self) -> bool {
178        self.to_maybe_accessor()
179            .map_or(false, |x| x.accessible_if.is_some())
180    }
181
182    /// Converts this to a MaybeAccessor,returning None if it is not the `Maybe` variant.
183    pub(crate) fn to_maybe_accessor(self) -> Option<MaybeAccessor<'a>> {
184        match self {
185            AccessorOrMaybe::Maybe(x) => Some(x),
186            _ => None,
187        }
188    }
189
190    #[allow(dead_code)]
191    pub(crate) fn is_maybe_accessor(&self) -> bool {
192        matches!(self, AccessorOrMaybe::Maybe { .. })
193    }
194}
195
196impl<'a> PrefixKind<'a> {
197    /// Gets the accessibility for a field,used for (very basic) runtime reflection.
198    pub(crate) fn field_accessor(&self, field: &Field<'a>) -> FieldAccessor<'a> {
199        use self::OnMissingField as OMF;
200
201        match self.fields[field] {
202            AccessorOrMaybe::Accessor => FieldAccessor::Method { name: None },
203            AccessorOrMaybe::Maybe(MaybeAccessor { on_missing, .. }) => match on_missing {
204                OMF::ReturnOption => FieldAccessor::MethodOption,
205                OMF::Panic { .. } | OMF::With { .. } | OMF::Value { .. } | OMF::Default_ => {
206                    FieldAccessor::Method { name: None }
207                }
208            },
209        }
210    }
211}

222/// Returns a value which for a prefix-type .
223pub(crate) fn prefix_type_tokenizer<'a>(
224    mono_type_layout: &'a Ident,
225    ds: &'a DataStructure<'a>,
226    config: &'a StableAbiOptions<'a>,
227    _ctokens: &'a CommonTokens<'a>,
228) -> Result<PrefixTypeTokens, syn::Error> {
229    if matches!(config.kind, StabilityKind::Prefix { .. }) {
230        if ds
231            .variants
232            .get(0)
233            .map_or(false, |struct_| struct_.fields.len() > 64)
234        {
235            return_spanned_err!(
236                ds.name,
237                "`#[sabi(kind(Prefix(..)))]` structs cannot have more than 64 fields."
238            );
239        }
240
241        if config.repr.is_packed.is_some() {
242            return_spanned_err!(
243                ds.name,
244                "`#[sabi(kind(Prefix(..)))]` structs cannot be `#[repr(C, packed)]`"
245            );
246        }
247    }
248
249    Ok({
250        fn default_prefixtype_tokens() -> PrefixTypeTokens {
251            PrefixTypeTokens {
252                prefixref_types: TokenStream2::new(),
253                prefixref_impls: TokenStream2::new(),
254            }
255        }
256
257        let struct_ = match ds.variants.get(0) {
258            Some(x) => x,
259            None => return Ok(default_prefixtype_tokens()),
260        };
261
262        let prefix = match &config.kind {
263            StabilityKind::Prefix(prefix) => prefix,
264            _ => return Ok(default_prefixtype_tokens()),
265        };
266
267        let first_suffix_field = prefix.first_suffix_field.field_pos;
268        let prefix_fields_struct = &prefix.prefix_fields_struct;
269
270        let doc_hidden_attr = config.doc_hidden_attr;
271
272        let deriving_name = ds.name;
273        let (ref impl_generics, ref ty_generics, ref where_clause) = ds.generics.split_for_impl();
274
275        let empty_preds = Punctuated::new();
276        let where_preds = where_clause
277            .as_ref()
278            .map_or(&empty_preds, |x| &x.predicates)
279            .into_iter();
280        let where_preds_a = where_preds.clone();
281        let where_preds_b = where_preds.clone();
282        let where_preds_c = where_preds.clone();
283        // let where_preds_d0=where_preds.clone();
284        // let where_preds_d1=where_preds.clone();
285        let where_preds_e = where_preds.clone();
286        let where_preds_f = where_preds.clone();
287        let where_preds_rl = where_preds.clone();
288        let where_preds_r2 = where_preds.clone();
289        let prefix_bounds = &prefix.prefix_bounds;
290
291        let stringified_deriving_name = deriving_name.to_string();
292
293        let stringified_generics = (&ty_generics).into_token_stream().to_string();
294        let stringified_generics_tokenizer = rstr_tokenizer(&stringified_generics);
295
296        let is_ds_pub = matches!(ds.vis, Visibility::Public { .. }) && doc_hidden_attr.is_none();
297        let has_replaced_docs = !prefix.replacing_prefix_ref_docs.is_empty();
298
299        let prefix_ref_docs = ToTokenFnMut::new(|ts| {
300            if !is_ds_pub {
301                return;
302            }
303
304            if has_replaced_docs {
305                let iter = prefix.replacing_prefix_ref_docs.iter();
306                ts.append_all(quote!(#(#[doc = #iter])*));
307            } else {
308                let single_docs = format!(
309                    "\
310                        This is the pointer to the prefix of \n\
311                        [`{deriving_name}{generics}`](struct@{deriving_name}).\n\
312                        \n\
313                        **This is automatically generated documentation,\
314                        by the StableAbi derive macro**.\n\
315                        \n\
316                        ### Creating a compiletime-constant\n\
317                        \n\
318                        You can look at the docs in [`::abi_stable::docs::prefix_types`] \
319                        to see how you\n\
320                        can construct and use this and similar types.<br>\n\
321                        More specifically in the \n\
322                        [\"constructing a module\" example\n\
323                        ](::abi_stable::docs::prefix_types#module_construction) or the\n\
324                        [\"Constructing a vtable\" example\n\
325                        ](::abi_stable::docs::prefix_types#vtable_construction)\n\
326                    ",
327                    deriving_name = stringified_deriving_name,
328                    generics = stringified_generics,
329                );
330
331                ts.append_all(quote!(#[doc = #single_docs]));
332            }
333        });
334
335        let prefix_fields_docs = if is_ds_pub {
336            format!(
337                "\
338This is the prefix fields of 
339[`{deriving_name}{generics}`](struct@{deriving_name}),
340accessible through [`{prefix_name}`](struct@{prefix_name}), with `.0.prefix()`.
341
342**This is automatically generated documentation,by the StableAbi derive macro**.
343                ",
344                prefix_name = prefix.prefix_ref,
345                deriving_name = stringified_deriving_name,
346                generics = stringified_generics,
347            )
348        } else {
349            String::new()
350        };
351
352        let prefix_ref = prefix.prefix_ref;
353
354        // Generating the `<prefix_ref>` struct
355        let generated_types = {
356            let vis = ds.vis;
357            let generics = ds.generics;
358
359            let alignemnt = if let Some(alignemnt) = config.repr.is_aligned {
360                let alignemnt = as_derive_utils::utils::uint_lit(alignemnt.into());
361                quote!(, align(#alignemnt))
362            } else {
363                quote!()
364            };
365
366            let prefix_field_iter = struct_.fields[..first_suffix_field].iter();
367            // This uses `field_<NUMBER>` for tuple fields
368            let prefix_field_vis = prefix_field_iter.clone().map(|f| {
369                ToTokenFnMut::new(move |ts| {
370                    let vis = f.vis;
371                    if AccessorOrMaybe::Accessor == prefix.fields[f] {
372                        vis.to_tokens(ts);
373                    }
374                })
375            });
376            let prefix_field = prefix_field_iter.clone().map(|f| f.pat_ident());
377            let prefix_field_ty = prefix_field_iter.clone().map(|f| f.ty);
378
379            let (_, ty_generics, where_clause) = generics.split_for_impl();
380
381            quote!(
382                #doc_hidden_attr
383                #prefix_ref_docs
384                #[repr(transparent)]
385                #vis struct #prefix_ref #generics (
386                    #vis ::abi_stable::pmr::PrefixRef<
387                        #prefix_fields_struct #ty_generics,
388                    >
389                )#where_clause;
390
391                #doc_hidden_attr
392                #[doc=#prefix_fields_docs]
393                // A struct with all the prefix fields in the deriving type
394                //
395                // A field being in the prefix doesn't mean that it's
396                // unconditionally accessible, it just means that it won't cause a SEGFAULT.
397                #[repr(C #alignemnt)]
398                #vis struct #prefix_fields_struct #generics
399                #where_clause
400                {
401                    #(
402                        ///
403                        #prefix_field_vis #prefix_field: #prefix_field_ty,
404                    )*
405                    // Using this to ensure:
406                    // - That all the generic arguments are used
407                    // - That the struct has the same alignemnt as the deriving struct.
408                    __sabi_pt_prefix_alignment: [#deriving_name #ty_generics  ;0],
409                    // Using this to ensure that the struct has at least the alignment of usize,
410                    // so that adding pointer fields is not an ABI breaking change.
411                    __sabi_usize_alignment: [usize; 0],
412                    __sabi_pt_unbounds: ::abi_stable::pmr::NotCopyNotClone,
413                }
414            )
415        };
416
417        let mut accessor_buffer = String::new();
418
419        let mut uncond_acc_docs = Vec::<String>::new();
420        let mut cond_acc_docs = Vec::<String>::new();
421        let mut field_index_for = Vec::new();
422        let offset_consts: &[syn::Ident] = &struct_
423            .fields
424            .iter()
425            .map(|f| parse_str_as_ident(&format!("__sabi_offset_for_{}", f.pat_ident())))
426            .collect::<Vec<Ident>>();
427
428        // Creates the docs for the accessor functions.
429        // Creates the identifiers for constants associated with each field.
430        for field in struct_.fields.iter() {
431            use self::AccessorOrMaybe as AOM;
432            use std::fmt::Write;
433
434            let mut acc_doc_buffer = String::new();
435            let acc_on_missing = prefix.fields[field];
436            if is_ds_pub {
437                let _ = write!(
438                    acc_doc_buffer,
439                    "Accessor method for the `{deriving_name}::{field_name}` field.",
440                    deriving_name = deriving_name,
441                    field_name = field.pat_ident(),
442                );
443
444                match acc_on_missing {
445                    AOM::Accessor => {
446                        acc_doc_buffer.push_str("This is for a field which always exists.")
447                    }
448                    AOM::Maybe(MaybeAccessor {
449                        on_missing: OnMissingField::ReturnOption,
450                        ..
451                    }) => acc_doc_buffer.push_str(
452                        "Returns `Some(field_value)` if the field exists,\
453                             `None` if it does not.\
454                            ",
455                    ),
456                    AOM::Maybe(MaybeAccessor {
457                        on_missing: OnMissingField::Panic,
458                        ..
459                    }) => acc_doc_buffer
460                        .push_str("\n\n# Panic\n\nPanics if the field does not exist."),
461                    AOM::Maybe(MaybeAccessor {
462                        on_missing: OnMissingField::With { function },
463                        ..
464                    }) => write!(
465                        acc_doc_buffer,
466                        "Returns `{function}()` if the field does not exist.",
467                        function = (&function).into_token_stream()
468                    )
469                    .drop_(),
470                    AOM::Maybe(MaybeAccessor {
471                        on_missing: OnMissingField::Value { .. },
472                        ..
473                    }) => acc_doc_buffer.push_str(
474                        "\
475                            Returns a default value (not Default::default()) \
476                            if the field does not exist.\
477                        ",
478                    ),
479                    AOM::Maybe(MaybeAccessor {
480                        on_missing: OnMissingField::Default_,
481                        ..
482                    }) => acc_doc_buffer
483                        .push_str("Returns `Default::default()` if the field does not exist."),
484                };
485            }
486
487            if config.with_field_indices {
488                let field_name = field.pat_ident();
489                let mut new_ident = parse_str_as_ident(&format!("field_index_for_{}", field_name));
490                new_ident.set_span(field_name.span());
491                field_index_for.push(new_ident);
492            }
493
494            match acc_on_missing {
495                AOM::Accessor => {
496                    uncond_acc_docs.push(acc_doc_buffer);
497                }
498                AOM::Maybe { .. } => {
499                    cond_acc_docs.push(acc_doc_buffer);
500                }
501            }
502        }
503
504        let mut unconditional_accessors = Vec::new();
505        let mut conditional_accessors = Vec::new();
506
507        // Creates TokenStreams for each accessor function.
508        for (field_i, field) in struct_.fields.iter().enumerate() {
509            use std::fmt::Write;
510            accessor_buffer.clear();
511            write!(accessor_buffer, "{}", field.pat_ident()).drop_();
512            let vis = field.vis;
513            let mut getter_name = syn::parse_str::<Ident>(&accessor_buffer).expect("BUG");
514            getter_name.set_span(field.pat_ident().span());
515            let field_name = field.pat_ident();
516            let field_span = field_name.span();
517            let ty = field.ty;
518
519            let accessor_bounds = &prefix.accessor_bounds[field];
520
521            let field_where_clause = if accessor_bounds.is_empty() {
522                None
523            } else {
524                Some(quote!(where #ty:))
525            };
526
527            match prefix.fields[field] {
528                AccessorOrMaybe::Accessor => {
529                    unconditional_accessors.push(quote_spanned! {field_span=>
530                        #[allow(clippy::missing_const_for_fn)]
531                        #vis fn #getter_name(&self)->#ty
532                        #field_where_clause #( #accessor_bounds+ )*
533                        {
534                            self.0.prefix().#field_name
535                        }
536                    })
537                }
538                AccessorOrMaybe::Maybe(maybe_accessor) => {
539                    let field_offset = &offset_consts[field_i];
540                    let on_missing_field = maybe_accessor.on_missing;
541                    let is_optional = on_missing_field == OnMissingField::ReturnOption;
542
543                    let return_ty = if is_optional {
544                        quote!( Option< #ty > )
545                    } else {
546                        quote!( #ty)
547                    };
548
549                    let else_ = match on_missing_field {
550                        OnMissingField::ReturnOption => quote_spanned! {field_span=>
551                            return None
552                        },
553                        OnMissingField::Panic => quote_spanned!(field_span=>
554                            __sabi_re::panic_on_missing_field_ty::<
555                                #deriving_name #ty_generics
556                            >(
557                                #field_i,
558                                self._prefix_type_layout(),
559                            )
560                        ),
561                        OnMissingField::With { function } => quote_spanned! {field_span=>
562                            #function()
563                        },
564                        OnMissingField::Value { value } => quote_spanned! {field_span=>
565                            (#value)
566                        },
567                        OnMissingField::Default_ => quote_spanned! {field_span=>
568                            Default::default()
569                        },
570                    };
571
572                    let val_var = syn::Ident::new("val", Span::mixed_site());
573
574                    let with_val = if is_optional {
575                        quote_spanned!(field_span=> Some(#val_var) )
576                    } else {
577                        val_var.to_token_stream()
578                    };
579
580                    conditional_accessors.push(quote_spanned! {field_span=>
581                        #[allow(clippy::missing_const_for_fn)]
582                        #vis fn #getter_name(&self)->#return_ty
583                        #field_where_clause #( #accessor_bounds+ )*
584                        {
585                            let acc_bits=self.0.field_accessibility().bits();
586                            let #val_var=if (1u64<<#field_i & Self::__SABI_PTT_FAM & acc_bits)==0 {
587                                #else_
588                            }else{
589                                unsafe{
590                                    *((self.0.to_raw_ptr() as *const u8)
591                                        .offset(Self::#field_offset as isize)
592                                        as *const #ty)
593                                }
594                            };
595                            #with_val
596                        }
597                    });
598                }
599            }
600        }
601
602        let cond_field_indices = &prefix.cond_field_indices;
603        let enable_field_if = &prefix.enable_field_if;
604        let unconditional_bit_mask = &prefix.unconditional_bit_mask;
605
606        let cond_field_indices = cond_field_indices.iter();
607        let enable_field_if = enable_field_if.iter();
608
609        let field_i_a = 0u8..;
610
611        let mut pt_layout_ident = parse_str_as_ident(&format!("__sabi_PT_LAYOUT{}", deriving_name));
612        pt_layout_ident.set_span(deriving_name.span());
613
614        let mut generated_impls = quote!(
615            #[allow(non_upper_case_globals)]
616            const #pt_layout_ident:&'static __sabi_re::PTStructLayout ={
617                &__sabi_re::PTStructLayout::new(
618                    #stringified_generics_tokenizer,
619                    #mono_type_layout,
620                )
621            };
622
623            unsafe impl #impl_generics
624                __sabi_re::PrefixTypeTrait
625            for #deriving_name #ty_generics
626            where
627                #(#where_preds_a,)*
628                #(#prefix_bounds,)*
629            {
630                // Describes the accessibility of all the fields,
631                // used to initialize the `WithMetadata<Self>::_prefix_type_field_acc` field.
632                const PT_FIELD_ACCESSIBILITY:__sabi_re::FieldAccessibility={
633                    __sabi_re::FieldAccessibility::from_u64(
634                        #unconditional_bit_mask
635                        #(
636                            |(((#enable_field_if)as u64) << #cond_field_indices)
637                        )*
638                    )
639                };
640                // A description of the struct used for error messages.
641                const PT_LAYOUT:&'static __sabi_re::PTStructLayout =#pt_layout_ident;
642
