5macro_rules! shared_fns {
6    (
7        erased=$erased:ident
8        unerased=$unerased:ident
9    ) => {
10        impl<T> $erased<T> {
11            pub(super) fn from_unerased_value(value: $unerased<T>) -> RBox<Self> {
12                unsafe {
13                    let boxed = RBox::new(value);
14                    ErasedType::from_unerased(boxed)
15                }
16            }
17
18            fn run<'a, F, R>(this: RRef<'a, Self>, f: F) -> R
19            where
20                F: FnOnce(&'a $unerased<T>) -> R,
21            {
22                extern_fn_panic_handling! {
23                    unsafe{
24                        f(Self::downcast_into(this).get())
25                    }
26                }
27            }
28
29            pub(super) extern "C" fn clone(this: RRef<'_, Self>) -> RBox<Self> {
30                Self::run(this, |this| Self::from_unerased_value(this.clone()))
31            }
32            pub(super) extern "C" fn is_empty(this: RRef<'_, Self>) -> bool {
33                Self::run(this, |this| this.is_empty())
34            }
35            pub(super) extern "C" fn is_full(this: RRef<'_, Self>) -> bool {
36                Self::run(this, |this| this.is_full())
37            }
38            pub(super) extern "C" fn len(this: RRef<'_, Self>) -> usize {
39                Self::run(this, |this| this.len())
40            }
41            pub(super) extern "C" fn capacity(this: RRef<'_, Self>) -> ROption<usize> {
42                Self::run(this, |this| this.capacity().into_c())
43            }
44        }
45    };
46}

58#[inline]
59fn rresult_from<T, E0, E1>(res: Result<T, E0>) -> RResult<T, E1>
60where
61    E0: Into<E1>,
62{
63    match res {
64        Ok(x) => ROk(x),
65        Err(e) => RErr(e.into()),
66    }
67}

69impl<T> ErasedSender<T> {
70    pub(super) extern "C" fn send(this: RRef<'_, Self>, val: T) -> RResult<(), RSendError<T>> {
71        Self::run(this, |this| this.send(val).piped(rresult_from))
72    }
73    pub(super) extern "C" fn try_send(
74        this: RRef<'_, Self>,
75        val: T,
76    ) -> RResult<(), RTrySendError<T>> {
77        Self::run(this, |this| this.try_send(val).piped(rresult_from))
78    }
79    pub(super) extern "C" fn send_timeout(
80        this: RRef<'_, Self>,
81        val: T,
82        timeout: RDuration,
83    ) -> RResult<(), RSendTimeoutError<T>> {
84        Self::run(this, |this| {
85            this.send_timeout(val, timeout.into()).piped(rresult_from)
86        })
87    }
88}
