15/// A late-initialized static reference,with fallible initialization.
16///
17/// As opposed to `Once`,
18/// this allows initialization of its static reference to happen fallibly,
19/// by returning a `Result<_,_>` from the `try_init` function,
20/// or by panicking inside either initialization function.
21///
22/// On `Err(_)` and panics,one can try initialializing the static reference again.
23///
24/// # Example
25///
26/// This lazily loads a configuration file.
27///
28/// ```
29///
30/// use abi_stable::{
31///     sabi_types::LateStaticRef,
32///     std_types::{RBox, RBoxError, RHashMap, RString},
33///     utils::leak_value,
34/// };
35///
36/// use std::{fs, io, path::Path};
37///
38/// use serde::Deserialize;
39///
40/// #[derive(Deserialize)]
41/// pub struct Config {
42///     pub user_actions: RHashMap<RString, UserAction>,
43/// }
44///
45/// #[derive(Deserialize)]
46/// pub enum UserAction {
47///     Include,
48///     Ignore,
49///     ReplaceWith,
50/// }
51///
52/// fn load_config(file_path: &Path) -> Result<&'static Config, RBoxError> {
53///     static CONFIG: LateStaticRef<&Config> = LateStaticRef::new();
54///
55///     CONFIG.try_init(|| {
56///         let file = load_file(file_path).map_err(RBoxError::new)?;
57///         let config =
58///             serde_json::from_str::<Config>(&file).map_err(RBoxError::new)?;
59///         Ok(leak_value(config))
60///     })
61/// }
62///
63/// # fn load_file(file_path:&Path)->Result<String,RBoxError>{
64/// #     let str=r##"
65/// #         {
66/// #             "user_actions":{
67/// #                 "oolong":"prolonged",
68/// #                 "genius":"idiot"
69/// #             }
70/// #         }
71/// #     "##.to_string();
72/// #     Ok(str)
73/// # }
74///
75/// ```
76///
77#[repr(C)]
78#[derive(StableAbi)]
79pub struct LateStaticRef<T> {
80    pointer: AtomicPtr<()>,
81    lock: RMutex<()>,
82    _marker: PhantomData<T>,
83}
84
85#[allow(clippy::declare_interior_mutable_const)]
86const LOCK: RMutex<()> = RMutex::new(());
87
88unsafe impl<T: Sync> Sync for LateStaticRef<T> {}
89unsafe impl<T: Send> Send for LateStaticRef<T> {}

91impl<T> LateStaticRef<T> {
92    /// Constructs the `LateStaticRef` in an uninitialized state.
93    ///
94    /// # Example
95    ///
96    /// ```
97    /// use abi_stable::sabi_types::LateStaticRef;
98    ///
99    /// static LATE_REF: LateStaticRef<&String> = LateStaticRef::new();
100    ///
101    /// ```
102    pub const fn new() -> Self {
103        Self {
104            lock: LOCK,
105            pointer: AtomicPtr::new(ptr::null_mut()),
106            _marker: PhantomData,
107        }
108    }

131impl<T> LateStaticRef<T> {
132    /// Constructs `LateStaticRef` from a [`PrefixRef`].
133    ///
134    /// # Example
135    ///
136    /// ```
137    /// use abi_stable::{
138    ///     pointer_trait::ImmutableRef,
139    ///     prefix_type::{PrefixRefTrait, PrefixTypeTrait, WithMetadata},
140    ///     sabi_types::LateStaticRef,
141    ///     StableAbi,
142    /// };
143    ///
144    /// fn main() {
145    ///     assert_eq!(LATE_REF.get().unwrap().get_number()(), 100);
146    /// }
147    ///
148    /// pub static LATE_REF: LateStaticRef<PersonMod_Ref> = {
149    ///     // This is how you can construct a `LateStaticRef<Foo_Ref>`,
150    ///     //  from a `Foo_Ref` at compile-time.
151    ///     //
152    ///     // If you don't need a `LateStaticRef` you can construct a `PersonMod_Ref` constant,
153    ///     // and use that.
154    ///     LateStaticRef::from_prefixref(MODULE.0)
155    /// };
156    ///
157    /// #[repr(C)]
158    /// #[derive(StableAbi)]
159    /// #[sabi(kind(Prefix))]
160    /// pub struct PersonMod {
161    ///     /// The `#[sabi(last_prefix_field)]` attribute here means that this is
162    ///     /// the last field in this struct that was defined in the
163    ///     /// first compatible version of the library.
164    ///     /// Moving this attribute is a braeking change.
165    ///     #[sabi(last_prefix_field)]
166    ///     pub get_number: extern "C" fn() -> u32,
167    /// }
168    ///
169    /// const MODULE: PersonMod_Ref = {
170    ///     const S: &WithMetadata<PersonMod> =
171    ///         &WithMetadata::new(PersonMod { get_number });
172    ///
173    ///     PersonMod_Ref(S.static_as_prefix())
174    /// };
175    ///
176    /// extern "C" fn get_number() -> u32 {
177    ///     100
178    /// }
179    /// ```
180    ///
181    /// [`PrefixRef`]: ../prefix_type/struct.PrefixRef.html
182    pub const fn from_prefixref(ptr: PrefixRef<T::PrefixFields>) -> Self
183    where
184        T: PrefixRefTrait + 'static,
185        T::PrefixFields: 'static,
186    {
187        Self {
188            lock: LOCK,
189            pointer: AtomicPtr::new(ptr.to_raw_ptr() as *mut ()),
190            _marker: PhantomData,
191        }
192    }
193}

195impl<T> LateStaticRef<T> {
196    /// Constructs `LateStaticRef` from a `NonNull` pointer.
197    ///
198    /// # Safety
199    ///
200    /// The passed in pointer must be valid for passing to
201    /// [`<T as ImmutableRef>::from_nonnull`],
202    /// it must be a valid pointer to `U`,
203    /// and be valid to dereference for the rest of the program's lifetime.
204    ///
205    /// [`<T as ImmutableRef>::from_nonnull`]:
206    /// ../pointer_trait/trait.ImmutableRef.html#method.from_nonnull
207    ///
208    /// # Example
209    ///
210    /// ```rust
211    /// use abi_stable::{
212    ///     pointer_trait::{GetPointerKind, PK_Reference},
213    ///     sabi_types::LateStaticRef,
214    ///     utils::ref_as_nonnull,
215    ///     StableAbi,
216    /// };
217    ///
218    /// use std::ptr::NonNull;
219    ///
220    /// #[derive(Copy, Clone)]
221    /// struct Foo<'a>(&'a u64);
222    ///
223    /// impl<'a> Foo<'a> {
224    ///     const fn as_nonnull(self) -> NonNull<u64> {
225    ///         ref_as_nonnull(self.0)
226    ///     }
227    /// }
228    ///
229    /// unsafe impl<'a> GetPointerKind for Foo<'a> {
230    ///     type PtrTarget = u64;
231    ///     type Kind = PK_Reference;
232    /// }
233    ///
234    /// const MODULE: LateStaticRef<Foo<'static>> = {
235    ///     unsafe {
236    ///         LateStaticRef::from_custom(Foo(&100).as_nonnull())
237    ///     }
238    /// };
239    /// ```
240    pub const unsafe fn from_custom(ptr: NonNull<T::PtrTarget>) -> Self
241    where
242        T: GetPointerKind<Kind = PK_Reference> + 'static,
243        T::PtrTarget: 'static,
244    {
245        Self {
246            lock: LOCK,
247            pointer: AtomicPtr::new(ptr.as_ptr() as *mut ()),
248            _marker: PhantomData,
249        }
250    }
251}

253impl<T> LateStaticRef<T>
254where
255    T: ImmutableRef + 'static,
256{
257    /// Lazily initializes the `LateStaticRef` with `initializer`,
258    /// returning the `T` if either it was already initialized,or
259    /// if `initalizer` returned Ok(..).
260    ///
261    /// If `initializer` returns an `Err(...)` this returns the error and
262    /// allows the `LateStaticRef` to be initializer later.
263    ///
264    /// If `initializer` panics,the panic is propagated,
265    /// and the reference can be initalized later.
266    ///
267    /// # Example
268    ///
269    /// ```
270    /// use abi_stable::{sabi_types::LateStaticRef, utils::leak_value};
271    ///
272    /// static LATE: LateStaticRef<&String> = LateStaticRef::new();
273    ///
274    /// static EARLY: LateStaticRef<&&str> = LateStaticRef::from_ref(&"Hello!");
275    ///
276    /// assert_eq!(LATE.try_init(|| Err("oh no!")), Err("oh no!"));
277    /// assert_eq!(
278    ///     LATE.try_init(|| -> Result<&'static String, ()> {
279    ///         Ok(leak_value("Yay".to_string()))
280    ///     })
281    ///     .map(|s| s.as_str()),
282    ///     Ok("Yay"),
283    /// );
284    ///
285    /// assert_eq!(EARLY.try_init(|| Err("oh no!")), Ok(&"Hello!"));
286    ///
287    ///
288    /// ```
289    pub fn try_init<F, E>(&self, initializer: F) -> Result<T, E>
290    where
291        F: FnOnce() -> Result<T, E>,
292    {
293        if let Some(pointer) = self.get() {
294            return Ok(pointer);
295        }
296
297        let guard_ = self.lock.lock();
298
299        if let Some(pointer) = self.get() {
300            return Ok(pointer);
301        }
302
303        let pointer = initializer()?;
304
305        self.pointer.store(
306            pointer.to_raw_ptr() as *mut T::PtrTarget as *mut (),
307            Ordering::Release,
308        );
309
310        drop(guard_);
311
312        Ok(pointer)
313    }
314
315    /// Lazily initializes the `LateStaticRef` with `initializer`,
316    /// returning the `T` if either it was already initialized,
317    /// or `initalizer` returns it without panicking.
318    ///
319    /// If `initializer` panics,the panic is propagated,
320    /// and the reference can be initalized later.
321    ///
322    /// # Example
323    ///
324    /// ```
325    /// use abi_stable::{sabi_types::LateStaticRef, utils::leak_value};
326    ///
327    /// static LATE: LateStaticRef<&String> = LateStaticRef::new();
328    ///
329    /// static EARLY: LateStaticRef<&&str> = LateStaticRef::from_ref(&"Hello!");
330    ///
331    /// let _ = std::panic::catch_unwind(|| {
332    ///     LATE.init(|| panic!());
333    /// });
334    ///
335    /// assert_eq!(LATE.init(|| leak_value("Yay".to_string())), &"Yay");
336    ///
337    /// assert_eq!(EARLY.init(|| panic!()), &"Hello!");
338    ///
339    /// ```
340    #[inline]
341    pub fn init<F>(&self, initializer: F) -> T
342    where
343        F: FnOnce() -> T,
344    {
345        self.try_init(|| -> Result<T, std::convert::Infallible> { Ok(initializer()) })
346            .expect("bug:LateStaticRef::try_init should only return an Err if `initializer` does")
347    }
348
349    /// Returns `Some(x:T)` if the `LateStaticRef` was initialized, otherwise returns `None`.
350    ///
351    /// # Example
352    ///
353    /// ```
354    /// use abi_stable::{sabi_types::LateStaticRef, utils::leak_value};
355    ///
356    /// static LATE: LateStaticRef<&String> = LateStaticRef::new();
357    ///
358    /// static EARLY: LateStaticRef<&&str> = LateStaticRef::from_ref(&"Hello!");
359    ///
360    /// let _ = std::panic::catch_unwind(|| {
361    ///     LATE.init(|| panic!());
362    /// });
363    ///
364    /// assert_eq!(LATE.get(), None);
365    /// LATE.init(|| leak_value("Yay".to_string()));
366    /// assert_eq!(LATE.get().map(|s| s.as_str()), Some("Yay"));
367    ///
368    /// assert_eq!(EARLY.get(), Some(&"Hello!"));
369    ///
370    /// ```
371    pub fn get(&self) -> Option<T> {
372        unsafe { T::from_raw_ptr(self.pointer.load(Ordering::Acquire) as *const T::PtrTarget) }
373    }
374}

383//#[cfg(test)]
384#[cfg(all(test, not(feature = "only_new_tests")))]
385mod tests {
386    use super::*;
387
388    use std::panic::catch_unwind;
389
390    static N_100: u32 = 100;
391    static N_277: u32 = 277;
392
393    #[test]
394    fn test_init() {
395        let ptr = LateStaticRef::<&u32>::new();
396
397        assert_eq!(None, ptr.get());
398
399        let caught = catch_unwind(|| {
400            ptr.init(|| panic!());
401        });
402        assert!(caught.is_err());
403
404        assert_eq!(None, ptr.get());
405
406        assert_eq!(100, *ptr.init(|| &N_100));
407        assert_eq!(100, *ptr.init(|| panic!("this should not run")));
408
409        assert_eq!((&N_100) as *const u32, ptr.get().unwrap() as *const u32);
410    }
411
412    #[test]
413    fn test_try_init() {
414        let ptr = LateStaticRef::<&u32>::new();
415
416        assert_eq!(None, ptr.get());
417
418        let caught = catch_unwind(|| {
419            let _ = ptr.try_init(|| -> Result<_, i32> { panic!() });
420        });
421        assert!(caught.is_err());
422
423        assert_eq!(None, ptr.get());
424
425        assert_eq!(Err(10), ptr.try_init(|| -> Result<_, i32> { Err(10) }));
426        assert_eq!(Err(17), ptr.try_init(|| -> Result<_, i32> { Err(17) }));
427
428        assert_eq!(Ok(&277), ptr.try_init(|| -> Result<_, i32> { Ok(&N_277) }));
429
430        assert_eq!(
431            Ok(&277),
432            ptr.try_init(|| -> Result<_, i32> { panic!("this should not run") })
433        );
434
435        assert_eq!((&N_277) as *const u32, ptr.get().unwrap() as *const u32);
436    }
