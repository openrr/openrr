8macro_rules! declare_iter_interface {
9    (
10        $k: ident => $v: ident;
11        $(#[$attr: meta])*
12        interface = $interface: ident;
13        type Item = $item: ty;
14    ) => (
15        #[repr(C)]
16        #[derive(StableAbi)]
17        $(#[$attr])*
18        pub struct $interface<$k, $v>(PhantomData<($k, $v)>);
19
20        impl<$k, $v> $interface<$k, $v>{
21            /// Constructs this type.
22            pub const NEW: Self = Self(PhantomData);
23        }
24
25
26        impl<'a, $k: 'a, $v: 'a> IteratorItem<'a> for $interface<$k, $v>{
27            type Item = $item;
28        }
29    )
30}

59type IntoIterInner<'a, K, V> = DynTrait<'a, RBox<()>, ValIterInterface<K, V>>;
60
61/// An iterator that yields all the entries of an `RHashMap`,
62/// deallocating the hashmap afterwards.
63///
64/// This implements `Iterator<Item= Tuple2< K, V > > + !Send + !Sync`
65#[repr(transparent)]
66#[derive(StableAbi)]
67pub struct IntoIter<K, V> {
68    iter: IntoIterInner<'static, u32, u32>,
69    _marker: PhantomData<(K, V, UnsafeIgnoredType<std::rc::Rc<()>>)>,
70}
71
72impl<K, V> IntoIter<K, V> {
73    /// # Safety
74    ///
75    /// This must be called only in `ErasedMap::into_val`.
76    pub(super) unsafe fn new<'a>(iter: DynTrait<'a, RBox<()>, ValIterInterface<K, V>>) -> Self
77    where
78        K: 'a,
79        V: 'a,
80    {
81        IntoIter {
82            iter: unsafe {
83                // SAFETY: the `'a` lifetime is erased because it's the lifetime of `K` and `V`,
84                // so it's implied by their usage.
85                mem::transmute::<IntoIterInner<'a, K, V>, IntoIterInner<'static, u32, u32>>(iter)
86            },
87            _marker: PhantomData,
88        }
89    }
90
91    #[inline]
92    fn iter(&self) -> &IntoIterInner<'_, K, V> {
93        unsafe { transmute_reference::<IntoIterInner<'static, u32, u32>, _>(&self.iter) }
94    }
95    #[inline]
96    fn iter_mut(&mut self) -> &mut IntoIterInner<'_, K, V> {
97        unsafe { transmute_mut_reference::<IntoIterInner<'static, u32, u32>, _>(&mut self.iter) }
98    }
99}
100
101impl<K, V> Iterator for IntoIter<K, V> {
102    type Item = Tuple2<K, V>;
103
104    #[inline]
105    fn next(&mut self) -> Option<Tuple2<K, V>> {
106        self.iter_mut().next()
107    }
108
109    #[inline]
110    fn nth(&mut self, nth: usize) -> Option<Tuple2<K, V>> {
111        self.iter_mut().nth(nth)
112    }
113
114    #[inline]
115    fn size_hint(&self) -> (usize, Option<usize>) {
116        self.iter().size_hint()
117    }
118
119    #[inline]
120    fn count(mut self) -> usize {
121        self.iter_mut().by_ref().count()
122    }
123
124    #[inline]
125    fn last(mut self) -> Option<Tuple2<K, V>> {
126        self.iter_mut().by_ref().last()
127    }
128}
