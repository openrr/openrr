1/// Reverse the first length bits of n.
2/// (Passing more than 16 as length will produce garbage.
3pub fn reverse_bits(mut n: u16, length: u8) -> u16 {
4    // Borrowed from http://aggregate.org/MAGIC/#Bit%20Reversal
5    n = ((n & 0xaaaa) >> 1) | ((n & 0x5555) << 1);
6    n = ((n & 0xcccc) >> 2) | ((n & 0x3333) << 2);
7    n = ((n & 0xf0f0) >> 4) | ((n & 0x0f0f) << 4);
8    n = ((n & 0xff00) >> 8) | ((n & 0x00ff) << 8);
9    n >> (16 - length)
10}
