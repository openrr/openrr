36def compute_intervals(zeros):
37    intervals = []
38    for i in range(0, len(zeros)):
39        if i == 0:
40            a = (zeros[i]) / 2 - 0.05 - zeros[i]
41            b = (zeros[i] + zeros[i + 1]) / 2 + 0.05 - zeros[i]
