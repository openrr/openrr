20/// Combined into one function (rather than two) to make it easier
21/// for the borrow checker.
22type GetSetValue<'a> = Box<dyn 'a + FnMut(Option<f64>) -> f64>;
23
24fn get(get_set_value: &mut GetSetValue<'_>) -> f64 {
25    (get_set_value)(None)
26}
27
28fn set(get_set_value: &mut GetSetValue<'_>, value: f64) {
29    (get_set_value)(Some(value));
30}

56/// Specifies how values in a [`Slider`] are clamped.
57#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
58#[cfg_attr(feature = "serde", derive(serde::Deserialize, serde::Serialize))]
59pub enum SliderClamping {
60    /// Values are not clamped.
61    ///
62    /// This means editing the value with the keyboard,
63    /// or dragging the number next to the slider will always work.
64    ///
65    /// The actual slider part is always clamped though.
66    Never,
67
68    /// Users cannot enter new values that are outside the range.
69    ///
70    /// Existing values remain intact though.
71    Edits,
72
73    /// Always clamp values, even existing ones.
74    #[default]
75    Always,
76}

78/// Control a number with a slider.
79///
80/// The slider range defines the values you get when pulling the slider to the far edges.
81/// By default all values are clamped to this range, even when not interacted with.
82/// You can change this behavior by passing `false` to [`Slider::clamp_to_range`].
83///
84/// The range can include any numbers, and go from low-to-high or from high-to-low.
85///
86/// The slider consists of three parts: a slider, a value display, and an optional text.
87/// The user can click the value display to edit its value. It can be turned off with `.show_value(false)`.
88///
89/// ```
90/// # egui::__run_test_ui(|ui| {
91/// # let mut my_f32: f32 = 0.0;
92/// ui.add(egui::Slider::new(&mut my_f32, 0.0..=100.0).text("My value"));
93/// # });
94/// ```
95///
96/// The default [`Slider`] size is set by [`crate::style::Spacing::slider_width`].
97#[must_use = "You should put this widget in a ui with `ui.add(widget);`"]
98pub struct Slider<'a> {
99    get_set_value: GetSetValue<'a>,
100    range: RangeInclusive<f64>,
101    spec: SliderSpec,
102    clamping: SliderClamping,
103    smart_aim: bool,
104    show_value: bool,
105    orientation: SliderOrientation,
106    prefix: String,
107    suffix: String,
108    text: WidgetText,
109
110    /// Sets the minimal step of the widget value
111    step: Option<f64>,
112
113    drag_value_speed: Option<f64>,
114    min_decimals: usize,
115    max_decimals: Option<usize>,
116    custom_formatter: Option<NumFormatter<'a>>,
117    custom_parser: Option<NumParser<'a>>,
118    trailing_fill: Option<bool>,
119    handle_shape: Option<HandleShape>,
120}

123    /// Creates a new horizontal slider.
124    ///
125    /// The `value` given will be clamped to the `range`,
126    /// unless you change this behavior with [`Self::clamping`].
127    pub fn new<Num: emath::Numeric>(value: &'a mut Num, range: RangeInclusive<Num>) -> Self {
128        let range_f64 = range.start().to_f64()..=range.end().to_f64();
129        let slf = Self::from_get_set(range_f64, move |v: Option<f64>| {
130            if let Some(v) = v {
131                *value = Num::from_f64(v);
132            }
133            value.to_f64()
134        });
135
136        if Num::INTEGRAL {
137            slf.integer()
138        } else {
139            slf
140        }
141    }

252    /// ### With `.clamping(SliderClamping::Always)` (default)
253    /// ```

486    /// Set `custom_formatter` and `custom_parser` to display and parse numbers as binary integers. Floating point
487    /// numbers are *not* supported.
488    ///
489    /// `min_width` specifies the minimum number of displayed digits; if the number is shorter than this, it will be
490    /// prefixed with additional 0s to match `min_width`.
491    ///
492    /// If `twos_complement` is true, negative values will be displayed as the 2's complement representation. Otherwise
493    /// they will be prefixed with a '-' sign.
494    ///
495    /// # Panics
496    ///
497    /// Panics if `min_width` is 0.
498    ///
499    /// ```
500    /// # egui::__run_test_ui(|ui| {
501    /// # let mut my_i32: i32 = 0;
502    /// ui.add(egui::Slider::new(&mut my_i32, -100..=100).binary(64, false));
503    /// # });
504    /// ```
505    pub fn binary(self, min_width: usize, twos_complement: bool) -> Self {
506        assert!(
507            min_width > 0,
508            "Slider::binary: `min_width` must be greater than 0"
509        );
510        if twos_complement {
511            self.custom_formatter(move |n, _| format!("{:0>min_width$b}", n as i64))
512        } else {
513            self.custom_formatter(move |n, _| {
514                let sign = if n < 0.0 { MINUS_CHAR_STR } else { "" };
515                format!("{sign}{:0>min_width$b}", n.abs() as i64)
516            })
517        }
518        .custom_parser(|s| i64::from_str_radix(s, 2).map(|n| n as f64).ok())
519    }

556    /// Set `custom_formatter` and `custom_parser` to display and parse numbers as hexadecimal integers. Floating point
557    /// numbers are *not* supported.
558    ///
559    /// `min_width` specifies the minimum number of displayed digits; if the number is shorter than this, it will be
560    /// prefixed with additional 0s to match `min_width`.
561    ///
562    /// If `twos_complement` is true, negative values will be displayed as the 2's complement representation. Otherwise
563    /// they will be prefixed with a '-' sign.

565    /// # Panics
566    ///
567    /// Panics if `min_width` is 0.
568    ///
569    /// ```
570    /// # egui::__run_test_ui(|ui| {
571    /// # let mut my_i32: i32 = 0;
572    /// ui.add(egui::Slider::new(&mut my_i32, -100..=100).hexadecimal(16, false, true));
573    /// # });
574    /// ```
575    pub fn hexadecimal(self, min_width: usize, twos_complement: bool, upper: bool) -> Self {
576        assert!(
577            min_width > 0,
578            "Slider::hexadecimal: `min_width` must be greater than 0"
579        );
580        match (twos_complement, upper) {
581            (true, true) => {
582                self.custom_formatter(move |n, _| format!("{:0>min_width$X}", n as i64))
583            }
584            (true, false) => {
585                self.custom_formatter(move |n, _| format!("{:0>min_width$x}", n as i64))
586            }
587            (false, true) => self.custom_formatter(move |n, _| {
588                let sign = if n < 0.0 { MINUS_CHAR_STR } else { "" };
589                format!("{sign}{:0>min_width$X}", n.abs() as i64)
590            }),
591            (false, false) => self.custom_formatter(move |n, _| {
592                let sign = if n < 0.0 { MINUS_CHAR_STR } else { "" };
593                format!("{sign}{:0>min_width$x}", n.abs() as i64)
594            }),
595        }
596        .custom_parser(|s| i64::from_str_radix(s, 16).map(|n| n as f64).ok())
597    }

618        }
619
620        if let Some(step) = self.step {
621            let start = *self.range.start();
622            value = start + ((value - start) / step).round() * step;
623        }
624        if let Some(max_decimals) = self.max_decimals {
625            value = emath::round_to_decimals(value, max_decimals);
626        }
627        set(&mut self.get_set_value, value);
628    }
629
630    fn range(&self) -> RangeInclusive<f64> {
631        self.range.clone()
632    }
633
634    /// For instance, `position` is the mouse position and `position_range` is the physical location of the slider on the screen.
635    fn value_from_position(&self, position: f32, position_range: Rangef) -> f64 {
636        let normalized = remap_clamp(position, position_range, 0.0..=1.0) as f64;
637        value_from_normalized(normalized, self.range(), &self.spec)
638    }
639
640    fn position_from_value(&self, value: f64, position_range: Rangef) -> f32 {
641        let normalized = normalized_from_value(value, self.range(), &self.spec);
642        lerp(position_range, normalized as f32)
643    }
644}

646impl<'a> Slider<'a> {
647    /// Just the slider, no text
648    fn allocate_slider_space(&self, ui: &mut Ui, thickness: f32) -> Response {
649        let desired_size = match self.orientation {
650            SliderOrientation::Horizontal => vec2(ui.spacing().slider_width, thickness),
651            SliderOrientation::Vertical => vec2(thickness, ui.spacing().slider_width),
652        };
653        ui.allocate_response(desired_size, Sense::drag())
654    }
655
656    /// Just the slider, no text
657    fn slider_ui(&mut self, ui: &Ui, response: &Response) {
658        let rect = &response.rect;
659        let handle_shape = self
660            .handle_shape
661            .unwrap_or_else(|| ui.style().visuals.handle_shape);
662        let position_range = self.position_range(rect, &handle_shape);
663
664        if let Some(pointer_position_2d) = response.interact_pointer_pos() {
665            let position = self.pointer_position(pointer_position_2d);
666            let new_value = if self.smart_aim {
667                let aim_radius = ui.input(|i| i.aim_radius());
668                emath::smart_aim::best_in_range_f64(
669                    self.value_from_position(position - aim_radius, position_range),
670                    self.value_from_position(position + aim_radius, position_range),
671                )
672            } else {
673                self.value_from_position(position, position_range)
674            };
675            self.set_value(new_value);
676        }
677
678        let mut decrement = 0usize;
679        let mut increment = 0usize;
680
681        if response.has_focus() {
682            ui.ctx().memory_mut(|m| {
683                m.set_focus_lock_filter(
684                    response.id,
685                    EventFilter {
686                        // pressing arrows in the orientation of the
687                        // slider should not move focus to next widget
688                        horizontal_arrows: matches!(
689                            self.orientation,
690                            SliderOrientation::Horizontal
691                        ),
692                        vertical_arrows: matches!(self.orientation, SliderOrientation::Vertical),
693                        ..Default::default()
694                    },
695                );
696            });
697
698            let (dec_key, inc_key) = match self.orientation {
699                SliderOrientation::Horizontal => (Key::ArrowLeft, Key::ArrowRight),
700                // Note that this is for moving the slider position,
701                // so up = decrement y coordinate:
702                SliderOrientation::Vertical => (Key::ArrowUp, Key::ArrowDown),
703            };
704
705            ui.input(|input| {
706                decrement += input.num_presses(dec_key);
707                increment += input.num_presses(inc_key);
708            });
709        }
710
711        #[cfg(feature = "accesskit")]
712        {
713            use accesskit::Action;
714            ui.input(|input| {
715                decrement += input.num_accesskit_action_requests(response.id, Action::Decrement);
716                increment += input.num_accesskit_action_requests(response.id, Action::Increment);
717            });
718        }
719
720        let kb_step = increment as f32 - decrement as f32;
721
722        if kb_step != 0.0 {
723            let ui_point_per_step = 1.0; // move this many ui points for each kb_step
724            let prev_value = self.get_value();
725            let prev_position = self.position_from_value(prev_value, position_range);
726            let new_position = prev_position + ui_point_per_step * kb_step;
727            let new_value = match self.step {
728                Some(step) => prev_value + (kb_step as f64 * step),
729                None if self.smart_aim => {
730                    let aim_radius = 0.49 * ui_point_per_step; // Chosen so we don't include `prev_value` in the search.
731                    emath::smart_aim::best_in_range_f64(
732                        self.value_from_position(new_position - aim_radius, position_range),
733                        self.value_from_position(new_position + aim_radius, position_range),
734                    )
735                }
736                _ => self.value_from_position(new_position, position_range),
737            };
738            self.set_value(new_value);
739        }
740
741        #[cfg(feature = "accesskit")]
742        {
743            use accesskit::{Action, ActionData};
744            ui.input(|input| {
745                for request in input.accesskit_action_requests(response.id, Action::SetValue) {
746                    if let Some(ActionData::NumericValue(new_value)) = request.data {
747                        self.set_value(new_value);
748                    }
749                }
750            });
751        }
752
753        // Paint it:
754        if ui.is_rect_visible(response.rect) {
755            let value = self.get_value();
756
757            let visuals = ui.style().interact(response);
758            let widget_visuals = &ui.visuals().widgets;
759            let spacing = &ui.style().spacing;
760
761            let rail_radius = (spacing.slider_rail_height / 2.0).at_least(0.0);
762            let rail_rect = self.rail_rect(rect, rail_radius);
763            let rounding = widget_visuals.inactive.rounding;
764
765            ui.painter()
766                .rect_filled(rail_rect, rounding, widget_visuals.inactive.bg_fill);
767
768            let position_1d = self.position_from_value(value, position_range);
769            let center = self.marker_center(position_1d, &rail_rect);
770
771            // Decide if we should add trailing fill.
772            let trailing_fill = self
773                .trailing_fill
774                .unwrap_or_else(|| ui.visuals().slider_trailing_fill);
775
776            // Paint trailing fill.
777            if trailing_fill {
778                let mut trailing_rail_rect = rail_rect;
779
780                // The trailing rect has to be drawn differently depending on the orientation.
781                match self.orientation {
782                    SliderOrientation::Horizontal => {
783                        trailing_rail_rect.max.x = center.x + rounding.nw;
784                    }
785                    SliderOrientation::Vertical => {
786                        trailing_rail_rect.min.y = center.y - rounding.se;
787                    }
788                };
789
790                ui.painter().rect_filled(
791                    trailing_rail_rect,
792                    rounding,
793                    ui.visuals().selection.bg_fill,
794                );
795            }
796
797            let radius = self.handle_radius(rect);
798
799            let handle_shape = self
800                .handle_shape
801                .unwrap_or_else(|| ui.style().visuals.handle_shape);
802            match handle_shape {
803                style::HandleShape::Circle => {
804                    ui.painter().add(epaint::CircleShape {
805                        center,
806                        radius: radius + visuals.expansion,
807                        fill: visuals.bg_fill,
808                        stroke: visuals.fg_stroke,
809                    });
810                }
811                style::HandleShape::Rect { aspect_ratio } => {
812                    let v = match self.orientation {
813                        SliderOrientation::Horizontal => Vec2::new(radius * aspect_ratio, radius),
814                        SliderOrientation::Vertical => Vec2::new(radius, radius * aspect_ratio),
815                    };
816                    let v = v + Vec2::splat(visuals.expansion);
817                    let rect = Rect::from_center_size(center, 2.0 * v);
818                    ui.painter()
819                        .rect(rect, visuals.rounding, visuals.bg_fill, visuals.fg_stroke);
820                }
821            }
822        }
823    }

825    fn marker_center(&self, position_1d: f32, rail_rect: &Rect) -> Pos2 {
826        match self.orientation {
827            SliderOrientation::Horizontal => pos2(position_1d, rail_rect.center().y),
828            SliderOrientation::Vertical => pos2(rail_rect.center().x, position_1d),
829        }
830    }
831
832    fn pointer_position(&self, pointer_position_2d: Pos2) -> f32 {
833        match self.orientation {
834            SliderOrientation::Horizontal => pointer_position_2d.x,
835            SliderOrientation::Vertical => pointer_position_2d.y,
836        }
837    }
838
839    fn position_range(&self, rect: &Rect, handle_shape: &style::HandleShape) -> Rangef {
840        let handle_radius = self.handle_radius(rect);
841        let handle_radius = match handle_shape {
842            style::HandleShape::Circle => handle_radius,
843            style::HandleShape::Rect { aspect_ratio } => handle_radius * aspect_ratio,
844        };
845        match self.orientation {
846            SliderOrientation::Horizontal => rect.x_range().shrink(handle_radius),
847            // The vertical case has to be flipped because the largest slider value maps to the
848            // lowest y value (which is at the top)
849            SliderOrientation::Vertical => rect.y_range().shrink(handle_radius).flip(),
850        }
851    }
852
853    fn rail_rect(&self, rect: &Rect, radius: f32) -> Rect {
854        match self.orientation {
855            SliderOrientation::Horizontal => Rect::from_min_max(
856                pos2(rect.left(), rect.center().y - radius),
857                pos2(rect.right(), rect.center().y + radius),
858            ),
859            SliderOrientation::Vertical => Rect::from_min_max(
860                pos2(rect.center().x - radius, rect.top()),
861                pos2(rect.center().x + radius, rect.bottom()),
862            ),
863        }
864    }
865
866    fn handle_radius(&self, rect: &Rect) -> f32 {
867        let limit = match self.orientation {
868            SliderOrientation::Horizontal => rect.height(),
869            SliderOrientation::Vertical => rect.width(),
870        };
871        limit / 2.5
872    }

874    fn value_ui(&mut self, ui: &mut Ui, position_range: Rangef) -> Response {
875        // If [`DragValue`] is controlled from the keyboard and `step` is defined, set speed to `step`
876        let change = ui.input(|input| {
877            input.num_presses(Key::ArrowUp) as i32 + input.num_presses(Key::ArrowRight) as i32
878                - input.num_presses(Key::ArrowDown) as i32
879                - input.num_presses(Key::ArrowLeft) as i32
880        });
881
882        let any_change = change != 0;
883        let speed = if let (Some(step), true) = (self.step, any_change) {
884            // If [`DragValue`] is controlled from the keyboard and `step` is defined, set speed to `step`
885            step
886        } else {
887            self.drag_value_speed
888                .unwrap_or_else(|| self.current_gradient(position_range))
889        };
890
891        let mut value = self.get_value();
892        let response = ui.add({
893            let mut dv = DragValue::new(&mut value)
894                .speed(speed)
895                .min_decimals(self.min_decimals)
896                .max_decimals_opt(self.max_decimals)
897                .suffix(self.suffix.clone())
898                .prefix(self.prefix.clone());
899
900            match self.clamping {
901                SliderClamping::Never => {}
902                SliderClamping::Edits => {
903                    dv = dv.range(self.range.clone()).clamp_existing_to_range(false);
904                }
905                SliderClamping::Always => {
906                    dv = dv.range(self.range.clone()).clamp_existing_to_range(true);
907                }
908            }
909
910            if let Some(fmt) = &self.custom_formatter {
911                dv = dv.custom_formatter(fmt);
912            };
913            if let Some(parser) = &self.custom_parser {
914                dv = dv.custom_parser(parser);
915            }
916            dv
917        });
918        if value != self.get_value() {
919            self.set_value(value);
920        }
921        response
922    }

924    /// delta(value) / delta(points)
925    fn current_gradient(&mut self, position_range: Rangef) -> f64 {
926        // TODO(emilk): handle clamping
927        let value = self.get_value();
928        let value_from_pos = |position: f32| self.value_from_position(position, position_range);
929        let pos_from_value = |value: f64| self.position_from_value(value, position_range);
930        let left_value = value_from_pos(pos_from_value(value) - 0.5);
931        let right_value = value_from_pos(pos_from_value(value) + 0.5);
932        right_value - left_value
933    }

935    fn add_contents(&mut self, ui: &mut Ui) -> Response {
936        let old_value = self.get_value();
937
938        if self.clamping == SliderClamping::Always {
939            self.set_value(old_value);
940        }
941
942        let thickness = ui
943            .text_style_height(&TextStyle::Body)
944            .at_least(ui.spacing().interact_size.y);
945        let mut response = self.allocate_slider_space(ui, thickness);
946        self.slider_ui(ui, &response);
947
948        let value = self.get_value();
949        response.changed = value != old_value;
950        response.widget_info(|| WidgetInfo::slider(ui.is_enabled(), value, self.text.text()));
951
952        #[cfg(feature = "accesskit")]
953        ui.ctx().accesskit_node_builder(response.id, |builder| {
954            use accesskit::Action;
955            builder.set_min_numeric_value(*self.range.start());
956            builder.set_max_numeric_value(*self.range.end());
957            if let Some(step) = self.step {
958                builder.set_numeric_value_step(step);
959            }
960            builder.add_action(Action::SetValue);
961
962            let clamp_range = if self.clamping == SliderClamping::Never {
963                f64::NEG_INFINITY..=f64::INFINITY
964            } else {
965                self.range()
966            };
967            if value < *clamp_range.end() {
968                builder.add_action(Action::Increment);
969            }
970            if value > *clamp_range.start() {
971                builder.add_action(Action::Decrement);
972            }
973        });
974
975        let slider_response = response.clone();
976
977        let value_response = if self.show_value {
978            let handle_shape = self
979                .handle_shape
980                .unwrap_or_else(|| ui.style().visuals.handle_shape);
981            let position_range = self.position_range(&response.rect, &handle_shape);
982            let value_response = self.value_ui(ui, position_range);
983            if value_response.gained_focus()
984                || value_response.has_focus()
985                || value_response.lost_focus()
986            {
987                // Use the [`DragValue`] id as the id of the whole widget,
988                // so that the focus events work as expected.
989                response = value_response.union(response);
990            } else {
991                // Use the slider id as the id for the whole widget
992                response = response.union(value_response.clone());
993            }
994            Some(value_response)
995        } else {
996            None
997        };
998
999        if !self.text.is_empty() {
1000            let label_response =
1001                ui.add(Label::new(self.text.clone()).wrap_mode(TextWrapMode::Extend));
1002            // The slider already has an accessibility label via widget info,
1003            // but sometimes it's useful for a screen reader to know
1004            // that a piece of text is a label for another widget,
1005            // e.g. so the text itself can be excluded from navigation.
1006            slider_response.labelled_by(label_response.id);
1007            if let Some(value_response) = value_response {
1008                value_response.labelled_by(label_response.id);
1009            }
1010        }
1011
1012        response
1013    }
1014}

1027// ----------------------------------------------------------------------------
1028// Helpers for converting slider range to/from normalized [0-1] range.
1029// Always clamps.
1030// Logarithmic sliders are allowed to include zero and infinity,
1031// even though mathematically it doesn't make sense.
1032
1033const INFINITY: f64 = f64::INFINITY;
1034
1035/// When the user asks for an infinitely large range (e.g. logarithmic from zero),
1036/// give a scale that this many orders of magnitude in size.
1037const INF_RANGE_MAGNITUDE: f64 = 10.0;
1038
1039fn value_from_normalized(normalized: f64, range: RangeInclusive<f64>, spec: &SliderSpec) -> f64 {
1040    let (min, max) = (*range.start(), *range.end());
1041
1042    if min.is_nan() || max.is_nan() {
1043        f64::NAN
1044    } else if min == max {
1045        min
1046    } else if min > max {
1047        value_from_normalized(1.0 - normalized, max..=min, spec)
1048    } else if normalized <= 0.0 {
1049        min
1050    } else if normalized >= 1.0 {
1051        max
1052    } else if spec.logarithmic {
1053        if max <= 0.0 {
1054            // non-positive range
1055            -value_from_normalized(normalized, -min..=-max, spec)
1056        } else if 0.0 <= min {
1057            let (min_log, max_log) = range_log10(min, max, spec);
1058            let log = lerp(min_log..=max_log, normalized);
1059            10.0_f64.powf(log)
1060        } else {
1061            assert!(min < 0.0 && 0.0 < max);
1062            let zero_cutoff = logarithmic_zero_cutoff(min, max);
1063            if normalized < zero_cutoff {
1064                // negative
1065                value_from_normalized(
1066                    remap(normalized, 0.0..=zero_cutoff, 0.0..=1.0),
1067                    min..=0.0,
1068                    spec,
1069                )
1070            } else {
1071                // positive
1072                value_from_normalized(
1073                    remap(normalized, zero_cutoff..=1.0, 0.0..=1.0),
1074                    0.0..=max,
1075                    spec,
1076                )
1077            }
1078        }
1079    } else {
1080        debug_assert!(
1081            min.is_finite() && max.is_finite(),
1082            "You should use a logarithmic range"
1083        );
1084        lerp(range, normalized.clamp(0.0, 1.0))
1085    }
1086}
1087
1088fn normalized_from_value(value: f64, range: RangeInclusive<f64>, spec: &SliderSpec) -> f64 {
1089    let (min, max) = (*range.start(), *range.end());
1090
1091    if min.is_nan() || max.is_nan() {
1092        f64::NAN
1093    } else if min == max {
1094        0.5 // empty range, show center of slider
1095    } else if min > max {
1096        1.0 - normalized_from_value(value, max..=min, spec)
1097    } else if value <= min {
1098        0.0
1099    } else if value >= max {
1100        1.0
1101    } else if spec.logarithmic {
1102        if max <= 0.0 {
1103            // non-positive range
1104            normalized_from_value(-value, -min..=-max, spec)
1105        } else if 0.0 <= min {
1106            let (min_log, max_log) = range_log10(min, max, spec);
1107            let value_log = value.log10();
1108            remap_clamp(value_log, min_log..=max_log, 0.0..=1.0)
1109        } else {
1110            assert!(min < 0.0 && 0.0 < max);
1111            let zero_cutoff = logarithmic_zero_cutoff(min, max);
1112            if value < 0.0 {
1113                // negative
1114                remap(
1115                    normalized_from_value(value, min..=0.0, spec),
1116                    0.0..=1.0,
1117                    0.0..=zero_cutoff,
1118                )
1119            } else {
1120                // positive side
1121                remap(
1122                    normalized_from_value(value, 0.0..=max, spec),
1123                    0.0..=1.0,
1124                    zero_cutoff..=1.0,
1125                )
1126            }
1127        }
1128    } else {
1129        debug_assert!(
1130            min.is_finite() && max.is_finite(),
1131            "You should use a logarithmic range"
1132        );
1133        remap_clamp(value, range, 0.0..=1.0)
1134    }
1135}
1136
1137fn range_log10(min: f64, max: f64, spec: &SliderSpec) -> (f64, f64) {
1138    assert!(spec.logarithmic);
1139    assert!(min <= max);
1140
1141    if min == 0.0 && max == INFINITY {
1142        (spec.smallest_positive.log10(), INF_RANGE_MAGNITUDE)
1143    } else if min == 0.0 {
1144        if spec.smallest_positive < max {
1145            (spec.smallest_positive.log10(), max.log10())
1146        } else {
1147            (max.log10() - INF_RANGE_MAGNITUDE, max.log10())
1148        }
1149    } else if max == INFINITY {
1150        if min < spec.largest_finite {
1151            (min.log10(), spec.largest_finite.log10())
1152        } else {
1153            (min.log10(), min.log10() + INF_RANGE_MAGNITUDE)
1154        }
1155    } else {
1156        (min.log10(), max.log10())
1157    }
1158}
1159
1160/// where to put the zero cutoff for logarithmic sliders
1161/// that crosses zero ?
1162fn logarithmic_zero_cutoff(min: f64, max: f64) -> f64 {
1163    assert!(min < 0.0 && 0.0 < max);
1164
1165    let min_magnitude = if min == -INFINITY {
1166        INF_RANGE_MAGNITUDE
1167    } else {
1168        min.abs().log10().abs()
1169    };
1170    let max_magnitude = if max == INFINITY {
1171        INF_RANGE_MAGNITUDE
1172    } else {
1173        max.log10().abs()
1174    };
1175
1176    let cutoff = min_magnitude / (min_magnitude + max_magnitude);
1177    debug_assert!(
1178        0.0 <= cutoff && cutoff <= 1.0,
1179        "Bad cutoff {cutoff:?} for min {min:?} and max {max:?}"
1180    );
1181    cutoff
1182}
