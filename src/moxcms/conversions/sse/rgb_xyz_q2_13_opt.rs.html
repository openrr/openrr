29use crate::conversions::rgbxyz_fixed::TransformMatrixShaperFixedPointOpt;
30use crate::conversions::sse::rgb_xyz::SseAlignedU16;
31use crate::conversions::sse::rgb_xyz_q2_13::_xmm_load_epi32;
32use crate::transform::PointeeSizeExpressible;
33use crate::{CmsError, Layout, TransformExecutor};
34use num_traits::AsPrimitive;
35#[cfg(target_arch = "x86")]
36use std::arch::x86::*;
37#[cfg(target_arch = "x86_64")]
38use std::arch::x86_64::*;
39
40pub(crate) struct TransformShaperQ2_13OptSse<
41    T: Copy,
42    const SRC_LAYOUT: u8,
43    const DST_LAYOUT: u8,
44    const LINEAR_CAP: usize,
45    const GAMMA_LUT: usize,
46    const PRECISION: i32,
47> {
48    pub(crate) profile: TransformMatrixShaperFixedPointOpt<i32, i16, T, LINEAR_CAP>,
49    pub(crate) bit_depth: usize,
50}
51
52impl<
53    T: Copy + PointeeSizeExpressible + 'static,
54    const SRC_LAYOUT: u8,
55    const DST_LAYOUT: u8,
56    const LINEAR_CAP: usize,
57    const GAMMA_LUT: usize,
58    const PRECISION: i32,
59> TransformShaperQ2_13OptSse<T, SRC_LAYOUT, DST_LAYOUT, LINEAR_CAP, GAMMA_LUT, PRECISION>
60where
61    u32: AsPrimitive<T>,
62{
63    #[target_feature(enable = "sse4.1")]
64    unsafe fn transform_impl(&self, src: &[T], dst: &mut [T]) -> Result<(), CmsError> {
65        let src_cn = Layout::from(SRC_LAYOUT);
66        let dst_cn = Layout::from(DST_LAYOUT);
67        let src_channels = src_cn.channels();
68        let dst_channels = dst_cn.channels();
69
70        let mut temporary = SseAlignedU16([0; 8]);
71
72        if src.len() / src_channels != dst.len() / dst_channels {
73            return Err(CmsError::LaneSizeMismatch);
74        }
75        if src.len() % src_channels != 0 {
76            return Err(CmsError::LaneMultipleOfChannels);
77        }
78        if dst.len() % dst_channels != 0 {
79            return Err(CmsError::LaneMultipleOfChannels);
80        }
81
82        let t = self.profile.adaptation_matrix.transpose();
83
84        let max_colors = ((1 << self.bit_depth) - 1).as_();
85
86        unsafe {
87            let m0 = _mm_setr_epi16(
88                t.v[0][0], t.v[1][0], t.v[0][1], t.v[1][1], t.v[0][2], t.v[1][2], 0, 0,
89            );
90            let m2 = _mm_setr_epi16(t.v[2][0], 1, t.v[2][1], 1, t.v[2][2], 1, 0, 0);
91
92            let rnd_val = ((1i32 << (PRECISION - 1)) as i16).to_ne_bytes();
93            let rnd = _mm_set1_epi32(i32::from_ne_bytes([0, 0, rnd_val[0], rnd_val[1]]));
94
95            let v_max_value = _mm_set1_epi32(GAMMA_LUT as i32 - 1);
96
97            for (src, dst) in src
98                .chunks_exact(src_channels)
99                .zip(dst.chunks_exact_mut(dst_channels))
100            {
101                let rp = &self.profile.linear[src[src_cn.r_i()]._as_usize()];
102                let gp = &self.profile.linear[src[src_cn.g_i()]._as_usize()];
103                let bp = &self.profile.linear[src[src_cn.b_i()]._as_usize()];
104
105                let mut r = _xmm_load_epi32(rp);
106                let mut g = _xmm_load_epi32(gp);
107                let mut b = _xmm_load_epi32(bp);
108                let a = if src_channels == 4 {
109                    src[src_cn.a_i()]
110                } else {
111                    max_colors
112                };
113
114                r = _mm_shuffle_epi32::<0>(r);
115                g = _mm_shuffle_epi32::<0>(g);
116                b = _mm_shuffle_epi32::<0>(b);
117
118                g = _mm_slli_epi32::<16>(g);
119
120                let zrg0 = _mm_or_si128(r, g);
121                let zbz0 = _mm_or_si128(b, rnd);
122
123                let v0 = _mm_madd_epi16(zrg0, m0);
124                let v1 = _mm_madd_epi16(zbz0, m2);
125
126                let mut v = _mm_add_epi32(v0, v1);
127
128                v = _mm_srai_epi32::<PRECISION>(v);
129                v = _mm_max_epi32(v, _mm_setzero_si128());
130                v = _mm_min_epi32(v, v_max_value);
131
132                _mm_store_si128(temporary.0.as_mut_ptr() as *mut _, v);
133
134                dst[dst_cn.r_i()] = self.profile.gamma[temporary.0[0] as usize];
135                dst[dst_cn.g_i()] = self.profile.gamma[temporary.0[2] as usize];
136                dst[dst_cn.b_i()] = self.profile.gamma[temporary.0[4] as usize];
137                if dst_channels == 4 {
138                    dst[dst_cn.a_i()] = a;
139                }
140            }
141        }
142
143        Ok(())
144    }
145}
146
147impl<
148    T: Copy + PointeeSizeExpressible + 'static + Default,
149    const SRC_LAYOUT: u8,
150    const DST_LAYOUT: u8,
151    const LINEAR_CAP: usize,
152    const GAMMA_LUT: usize,
153    const PRECISION: i32,
154> TransformExecutor<T>
155    for TransformShaperQ2_13OptSse<T, SRC_LAYOUT, DST_LAYOUT, LINEAR_CAP, GAMMA_LUT, PRECISION>
156where
157    u32: AsPrimitive<T>,
158{
159    fn transform(&self, src: &[T], dst: &mut [T]) -> Result<(), CmsError> {
160        unsafe { self.transform_impl(src, dst) }
161    }
