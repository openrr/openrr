22/// Decodes raw bytes using the deserializer encoding.
23/// The method will borrow if encoding is UTF-8 compatible and `name` contains
24/// only UTF-8 compatible characters (usually only ASCII characters).
25#[inline]
26fn decode_name<'n>(name: QName<'n>, decoder: Decoder) -> Result<Cow<'n, str>, DeError> {
27    let local = name.local_name();
28    Ok(decoder.decode(local.into_inner())?)
29}

31/// A deserializer for xml names of elements and attributes.
32///
33/// Used for deserializing values from:
34/// - attribute names (`<... name="..." ...>`)
35/// - element names (`<name>...</name>`)
36///
37/// Converts a name to an identifier string using the following rules:
38///
39/// - if it is an [`attribute`] name, put `@` in front of the identifier
40/// - if it is a namespace binding (`xmlns` or `xmlns:xxx`) put the decoded name
41///   to the identifier
42/// - put the decoded [`local_name()`] of a name to the identifier
43///
44/// The final identifier looks like `[@]local_name`, or `@xmlns`, or `@xmlns:binding`
45/// (where `[]` means optional element).
46///
47/// The deserializer also supports deserializing names as other primitive types:
48/// - numbers
49/// - booleans
50/// - unit (`()`) and unit structs
51/// - unit variants of the enumerations
52///
53/// Because `serde` does not define on which side type conversion should be
54/// performed, and because [`Deserialize`] implementation for that primitives
55/// in serde does not accept strings, the deserializer will perform conversion
56/// by itself.
57///
58/// The deserializer is able to deserialize unit and unit structs, but any name
59/// will be converted to the same unit instance. This is asymmetry with a serializer,
60/// which not able to serialize those types, because empty names are impossible
61/// in XML.
62///
63/// `deserialize_any()` returns the same result as `deserialize_identifier()`.
64///
65/// # Lifetimes
66///
67/// - `'i`: lifetime of the data that the deserializer borrows from the parsed input
68/// - `'d`: lifetime of a deserializer that holds a buffer with content of events
69///
70/// [`attribute`]: Self::from_attr
71/// [`local_name()`]: QName::local_name
72/// [`Deserialize`]: serde::Deserialize
73pub struct QNameDeserializer<'i, 'd> {
74    name: CowRef<'i, 'd, str>,
75}

77impl<'i, 'd> QNameDeserializer<'i, 'd> {
78    /// Creates deserializer from name of an attribute
79    pub fn from_attr(
80        name: QName<'d>,
81        decoder: Decoder,
82        key_buf: &'d mut String,
83    ) -> Result<Self, DeError> {
84        key_buf.clear();
85        key_buf.push('@');
86
87        // https://github.com/tafia/quick-xml/issues/537
88        // Namespace bindings (xmlns:xxx) map to `@xmlns:xxx` instead of `@xxx`
89        if name.as_namespace_binding().is_some() {
90            decoder.decode_into(name.into_inner(), key_buf)?;
91        } else {
92            let local = name.local_name();
93            decoder.decode_into(local.into_inner(), key_buf)?;
94        };
95
96        Ok(Self {
97            name: CowRef::Slice(key_buf),
98        })
99    }
100
101    /// Creates deserializer from name of an element
102    pub fn from_elem(name: CowRef<'i, 'd, [u8]>, decoder: Decoder) -> Result<Self, DeError> {
103        let local = match name {
104            CowRef::Input(borrowed) => match decode_name(QName(borrowed), decoder)? {
105                Cow::Borrowed(borrowed) => CowRef::Input(borrowed),
106                Cow::Owned(owned) => CowRef::Owned(owned),
107            },
108            CowRef::Slice(borrowed) => match decode_name(QName(borrowed), decoder)? {
109                Cow::Borrowed(borrowed) => CowRef::Slice(borrowed),
110                Cow::Owned(owned) => CowRef::Owned(owned),
111            },
112            CowRef::Owned(owned) => match decode_name(QName(&owned), decoder)? {
113                // SAFETY: Because result is borrowed, no changes was done
114                // and we can safely unwrap here
115                Cow::Borrowed(_) => CowRef::Owned(String::from_utf8(owned).unwrap()),
116                Cow::Owned(owned) => CowRef::Owned(owned),
117            },
118        };
119
120        Ok(Self { name: local })
121    }
122}

135    /// According to the <https://www.w3.org/TR/xmlschema11-2/#boolean>,
136    /// valid boolean representations are only `"true"`, `"false"`, `"1"`,
137    /// and `"0"`. But this method also handles following:
138    ///
139    /// |`bool` |XML content
140    /// |-------|-------------------------------------------------------------
141    /// |`true` |`"True"`,  `"TRUE"`,  `"t"`, `"Yes"`, `"YES"`, `"yes"`, `"y"`
142    /// |`false`|`"False"`, `"FALSE"`, `"f"`, `"No"`,  `"NO"`,  `"no"`,  `"n"`
143    fn deserialize_bool<V>(self, visitor: V) -> Result<V::Value, Self::Error>
144    where
145        V: Visitor<'de>,
146    {
147        str2bool(self.name.as_ref(), visitor)
148    }

188    /// Forwards deserialization to the [`Self::deserialize_identifier`]
189    #[inline]
190    fn deserialize_any<V>(self, visitor: V) -> Result<V::Value, Self::Error>
191    where
192        V: Visitor<'de>,
193    {
194        self.deserialize_identifier(visitor)
195    }
196
197    /// If `name` is an empty string then calls [`Visitor::visit_none`],
198    /// otherwise calls [`Visitor::visit_some`] with itself
199    fn deserialize_option<V>(self, visitor: V) -> Result<V::Value, Self::Error>
200    where
201        V: Visitor<'de>,
202    {
203        if self.name.is_empty() {
204            visitor.visit_none()
205        } else {
206            visitor.visit_some(self)
207        }
208    }
209
210    fn deserialize_newtype_struct<V>(
211        self,
212        _name: &'static str,
213        visitor: V,
214    ) -> Result<V::Value, Self::Error>
215    where
216        V: Visitor<'de>,
217    {
218        visitor.visit_newtype_struct(self)
219    }
220
221    /// Calls a [`Visitor::visit_str`] if [`name`] contains only UTF-8
222    /// compatible encoded characters and represents an element name and
223    /// a [`Visitor::visit_string`] in all other cases.
224    ///
225    /// [`name`]: Self::name
226    fn deserialize_identifier<V>(self, visitor: V) -> Result<V::Value, Self::Error>
227    where
228        V: Visitor<'de>,
229    {
230        match self.name {
231            CowRef::Input(name) => visitor.visit_borrowed_str(name),
232            CowRef::Slice(name) => visitor.visit_str(name),
233            CowRef::Owned(name) => visitor.visit_string(name),
234        }
235    }
236
237    fn deserialize_enum<V>(
238        self,
239        _name: &str,
240        _variants: &'static [&'static str],
241        visitor: V,
242    ) -> Result<V::Value, Self::Error>
243    where
244        V: Visitor<'de>,
245    {
246        visitor.visit_enum(self)
247    }
248}
