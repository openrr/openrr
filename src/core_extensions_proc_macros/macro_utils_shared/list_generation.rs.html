33// All the finite lists should be `List::List`
34pub(crate) enum List {
35    List(TokenStream, Spans),
36    RangeFrom(usize, Spans),
37    GenIdentRange(GenIdentRange),
38    Chain{
39        bounded: TokenStream,
40        spans: Spans,
41        unbounded: Box<List>,
42    },
43    Cycle(TokenStream, Spans),
44    Skip(usize, Box<List>),
45}
46
47
48impl List {
49    #[allow(dead_code)]
50    pub(crate) fn spans(&self) -> Spans {
51        match self {
52            Self::List(_, x) | Self::RangeFrom(_, x) => *x,
53            Self::GenIdentRange(gir) => {
54                let s = gir.span();
55                Spans{start: s, end: s}
56            },
57            Self::Chain{spans, ..} => *spans,
58            Self::Cycle(_, x) => *x,
59            Self::Skip(_, x) => x.spans(),
60        }
61    }
62    pub(crate) fn is_finite(&self) -> bool {
63        mmatches!(self, Self::List{..})
64    }
65}

70struct ParseManyLists<C>{
71    iter: Peekable<IntoIter>,
72    _marker: PhantomData<C>,
73}
74
75impl<C> Iterator for ParseManyLists<C> 
76where 
77    C: Constructors
78{
79    type Item = crate::Result<C::This>;
80    fn next(&mut self) -> Option<Self::Item> {
81        if let Some(_) = self.iter.peek() {
82            Some(parse_impl::<_, C>(&mut self.iter))
83        } else {
84            None
85        }
86    }
87}

92fn parse_impl<I, C>(iter: &mut I) -> crate::Result<C::This>
93where
94    I: Iterator<Item = TokenTree>,
95    C: Constructors,
96{
97    macro_rules! list_functions {
98        (
99            $ident:ident,
100            $group:ident,
101            $stream_iter:ident,
102            $fname:literal => $fblock:block
103            $( $name:literal => $block:block )* 
104        ) => {
105            macro_rules! method_names {
106                () => {
107                    concat!(
108                        "one of ",
109                        "`", $fname, "`",
110                        $(", `", $name, "`",)*
111                    )
112                };
113            }
114
115            const PARAM_MSG: &str = concat!("expected ", method_names!(),", or parentheses.");
116
117            match_token!{PARAM_MSG, iter.next() =>
118                Some(TokenTree::Ident($ident)) => {
119                    let keyword = $ident.to_string();
120                    
121                    const IDENT_ERR: &str =
122                        concat!("expected ", method_names!(),", or parentheses.");
123
124                    let paren_res = parse_parentheses(iter);
125
126                    match &keyword[..] {
127                        $fname => {
128                            let $group = try_!(paren_res);
129                            let $stream_iter = $group.stream().into_iter();
130
131                            $fblock
132                        }
133                        $(
134                            $name => {
135                                let $group = try_!(paren_res);
136                                #[allow(unused_mut)]
137                                let mut $stream_iter = $group.stream().into_iter();
138
139                                $block
140                            }
141                        )*
142                        other => {
143                            let err = format!("{}\nFound {}", IDENT_ERR, other);
144                            return Err(crate::Error::one_tt($ident.span(), &err));
145                        }
146                    }
147                }
148                Some(TokenTree::Group(group))  => {
149                    Ok(C::make_group(group.stream(), Spans::new(group.span(), group.span())))
150                }
151            }
152        };
153    }
154
155
156    list_functions!{
157        ident,
158        group,
159        stream,
160        "cycle" => {
161            let mut args = stream.into_iter();
162            let tokens = try_!(parse_bounded(&mut args));
163
164            try_!(expect_no_tokens(args));
165            C::make_cycle(tokens.stream(), Spans::new(ident.span(), group.span()))
166        }
167        "repeat" => {
168            let CountAnd{count, other: tokens,..} =
169                try_!(parse_count_and(stream, parse_bounded));
170
171            let tokens = if count == 0 {
172                TokenStream::new()
173            } else {
174                let repeated = tokens.stream().into_iter();
175                RepeatTimes::new(count, repeated).collect()
176            };
177
178            Ok(C::make_group(tokens, Spans::new(ident.span(), group.span())))
179        }
180        "take" => {
181            let CountAnd{count, other: tokens,..} =
182                try_!(parse_count_and(stream, parse_unbounded));
183
184            let tokens = tokens.into_iter().take(count).collect();
185
186            Ok(C::make_group(tokens, Spans::new(ident.span(), group.span())))
187        }
188        "skip" => {
189            let CountAnd{count, other,..} = try_!(parse_count_and(stream, parse_unbounded));
190
191            if other.is_finite() {
192                let spans = other.spans();
193                Ok(C::make_group(other.into_iter().skip(count).collect(), spans))
194            } else {
195                C::make_skip_unbounded(count, other)
196            }
197        }
198        "chain" => {
199            let iter = ParseManyLists{
200                iter: stream.into_iter().peekable(),
201                _marker: PhantomData,
202            };
203            
204            C::make_chain(iter, Spans::new(ident.span(), group.span()))
205        }
206        "gen_ident_range" => {
207            let range = try_!(gen_ident_range_just_idents(
208                &mut stream.peekable(),
209                parse_unbounded_range_param,
210            ));
211
212            let spans = Spans::new(ident.span(), group.span());
213            if range.is_unbounded() {
214                C::make_gen_idents_range_unb(range, spans)
215            } else {
216                Ok(C::make_group(range.collect(), spans))
217            }            
218        }
219        "range" => {
220            let mut stream = stream.peekable();
221            let rangeb = try_!(parse_range_param(&mut stream));
222
223            if let Some(rend) = rangeb.end {
224                let tokens = (rangeb.start..rend)
225                    .map(|i| usize_tt(i, rangeb.spans.start) )
226                    .collect::<TokenStream>();
227                Ok(C::make_group(tokens, Spans::new(ident.span(), group.span())))
228            } else {
229                C::make_range_start(rangeb)
230            }
231        }
232    }
233}

236trait Constructors: Sized {
237    type This;
238
239    fn make_cycle(ts: TokenStream, span: Spans) -> crate::Result<Self::This>;
240
241    fn make_skip_unbounded(count: usize, what: List) -> crate::Result<Self::This>;
242    
243    fn make_chain(_: ParseManyLists<Self>, span: Spans) -> crate::Result<Self::This>;
244    
245    fn make_gen_idents_range_unb(range: GenIdentRange, span: Spans) -> crate::Result<Self::This>;
246
247    fn make_group(ts: TokenStream, span: Spans) -> Self::This;
248
249    fn make_range_start(rangeb: RangeB) -> crate::Result<Self::This>;
250}
251
252struct Unbounded;
253
254impl Constructors for Unbounded {
255    type This = List;
256    
257    fn make_cycle(ts: TokenStream, spans: Spans) -> crate::Result<Self::This> {
258        Ok(List::Cycle(ts, spans))
259    }
260
261    fn make_skip_unbounded(count: usize, what: List) -> crate::Result<Self::This> {
262        Ok(List::Skip(count, Box::new(what)))
263    }
264
265    fn make_chain(iter: ParseManyLists<Self>, mut spans: Spans) -> crate::Result<Self::This> {
266        let mut bounded = TokenStream::new();
267        let mut unbounded = None::<Box<List>>;
268
269        for elem in iter {
270            let elem = try_!(elem);
271
272            spans.end = elem.spans().end;
273
274            if let None = unbounded {
275                if let List::List(list, _) = elem {
276                    bounded.extend(list);
277                } else {
278                    unbounded = Some(Box::new(elem))
279                }
280            }
281        }
282
283        if let Some(unbounded) = unbounded {
284            Ok(List::Chain{bounded, spans, unbounded})
285        } else {
286            Ok(List::List(bounded, spans))
287        }
288    }
289
290    fn make_gen_idents_range_unb(range: GenIdentRange, _: Spans) -> crate::Result<Self::This> {
291        Ok(List::GenIdentRange(range))
292    }
293
294    fn make_group(ts: TokenStream, spans: Spans) -> Self::This {
295        List::List(ts, spans)
296    }
297
298    fn make_range_start(r: RangeB) -> crate::Result<Self::This> {
299        Ok(List::RangeFrom(r.start, r.spans))
300    }
301}

304struct Bounded;
305
306impl Constructors for Bounded {
307    type This = Group;
308
309    fn make_cycle(_: TokenStream, spans: Spans) -> crate::Result<Self::This> {
310        Err(crate::Error::with_spans(spans, "expected a bounded iterator"))
311    }
312
313    fn make_skip_unbounded(_: usize, what: List) -> crate::Result<Self::This> {
314        Err(crate::Error::with_spans(what.spans(), "expected a bounded iterator"))
315    }
316
317    fn make_chain(iter: ParseManyLists<Self>, span: Spans) -> crate::Result<Self::This> {
318        let mut tokens = TokenStream::new();
319
320        for elem in iter {
321            let elem = try_!(elem);
322            tokens.extend(elem.stream());
323        }
324
325        let mut group = Group::new(Delimiter::Parenthesis, tokens);
326        group.set_span(span.start);
327        Ok(group)
328    }
329    
330    fn make_gen_idents_range_unb(_: GenIdentRange, spans: Spans) -> crate::Result<Self::This> {
331        Err(crate::Error::with_spans(spans, "expected a bounded range"))
332    }
333
334    fn make_group(ts: TokenStream, span: Spans) -> Self::This {
335        let mut group = Group::new(Delimiter::Parenthesis, ts);
336        group.set_span(span.start);
337        group
338    }
339
340    fn make_range_start(r: RangeB) -> crate::Result<Self::This> {
341        Err(crate::Error::with_spans(r.spans, "Expected a bounded range"))
342    }
343}

361pub(crate) enum ListIter {
362    List(IntoIter),
363    RangeFrom(RangeFrom<usize>, Spans),
364    GenIdentRange(GenIdentRange),
365    Chain(Chain<IntoIter, Box<ListIter>>),
366    Cycle(Cycle<IntoIter>),
367}
368
369
370impl IntoIterator for List {
371    type Item = TokenTree;
372    type IntoIter = ListIter;
373
374    fn into_iter(self) -> ListIter {
375        match self {
376            Self::List(ts, _) => ListIter::List(ts.into_iter()),
377            Self::RangeFrom(start, span) => ListIter::RangeFrom(start.., span),
378            Self::GenIdentRange(gir) => ListIter::GenIdentRange(gir),
379            Self::Chain{bounded, unbounded, ..} =>
380                ListIter::Chain(bounded.into_iter().chain(Box::new(unbounded.into_iter()))),
381            Self::Cycle(x, _) => ListIter::Cycle(x.into_iter().cycle()),
382            Self::Skip(n, this) => {
383                let mut iter = (*this).into_iter();
384                if n != 0 { iter.nth(n-1); }
385                iter
386            },
387        }
388    }
389}
390
391impl ListIter {
392    #[allow(dead_code)]
393    pub(crate) fn is_finite(&self) -> bool {
394        mmatches!(self, Self::List{..})
395    }
396}
397
398impl Iterator for ListIter{
399    type Item = TokenTree;
400
401    fn next(&mut self) -> Option<TokenTree> {
402        match self {
403            Self::List(x) => x.next(),
404            Self::RangeFrom(x, span) => x.next().map(|x| usize_tt(x, span.start) ),
405            Self::GenIdentRange(x) => x.next(),
406            Self::Chain(x) => x.next(),
407            Self::Cycle(x) => x.next(),
408        }
409    }
410}
