3macro_rules! declare_multi_tl_types {(
4    attrs=[ $($extra_attrs:meta),* $(,)* ]
5) => (
6
7    /// A range of indices into a slice of type layouts
8    /// which can store up to five indices inline,
9    /// requiring additional layouts to be stored contiguously after the
10    /// fourth one in the slice.
11    #[repr(C)]
12    #[derive(Debug, Copy, Clone, PartialEq, Eq, Ord, PartialOrd)]
13    $(#[ $extra_attrs ])*
14    pub struct TypeLayoutRange{
15        bits0:u32,
16        bits1:u32,
17    }
18
19    impl TypeLayoutRange{
20        /// An empty `TypeLayoutRange`.
21        pub const EMPTY:Self=Self{
22            bits0:0,
23            bits1:0,
24        };
25
26        const LEN_MASK:u32=0b11_1111_1111;
27        const INDEX_MASK:u32=0x3FF;
28
29        const LEN_BIT_SIZE:u32=10;
30
31        /// How many indices can be stored inline.
32        pub const STORED_INLINE:usize=5;
33
34        const INDEX_0_OFFSET:u32=Self::LEN_BIT_SIZE;
35        const INDEX_1_OFFSET:u32=Self::LEN_BIT_SIZE+TypeLayoutIndex::BIT_SIZE;
36
37        const INDEX_2_OFFSET:u32=0;
38        const INDEX_3_OFFSET:u32=TypeLayoutIndex::BIT_SIZE;
39        const INDEX_4_OFFSET:u32=TypeLayoutIndex::BIT_SIZE*2;
40
41        fn size_assertions(){
42            let _:[(); 32-(Self::LEN_BIT_SIZE+TypeLayoutIndex::BIT_SIZE*2)as usize ];
43        }
44
45        /// Constructs a `TypeLayoutRange` with up to five type layout indices.
46        #[inline]
47        pub const fn with_up_to_5(mut len:usize,indices:[u16;5]) -> Self {
48            let len=len & 0usize.wrapping_sub((len <= Self::STORED_INLINE) as usize);
49            Self::with_more_than_5(len,indices)
50        }
51
52        /// Constructs a `TypeLayoutRange` with more
53        /// than five type layout indices,
54        /// in which the indices from `indices[4]` onwards are
55        /// stored contiguously in the slice.
56        #[inline]
57        pub const fn with_more_than_5(len:usize,indices:[u16;5]) -> Self {
58            Self{
59                bits0:len as u32
60                    |((indices[0] as u32 & Self::INDEX_MASK) << Self::INDEX_0_OFFSET)
61                    |((indices[1] as u32 & Self::INDEX_MASK) << Self::INDEX_1_OFFSET),
62                bits1:
63                     ((indices[2] as u32 & Self::INDEX_MASK) << Self::INDEX_2_OFFSET)
64                    |((indices[3] as u32 & Self::INDEX_MASK) << Self::INDEX_3_OFFSET)
65                    |((indices[4] as u32 & Self::INDEX_MASK) << Self::INDEX_4_OFFSET),
66            }
67        }
68
69        /// Constructs this `TypeLayoutRange` from its representation.
70        #[inline]
71        pub const fn from_u64(bits:u64) -> Self {
72            Self{
73                bits0:bits as u32,
74                bits1:(bits>>32)as u32,
75            }
76        }
77
78        /// Converts this `TypeLayoutRange` into its representation.
79        #[inline]
80        pub const fn to_u64(&self) -> u64 {
81             (self.bits0 as u64)
82            |((self.bits1 as u64) << 32)
83        }
84
85        /// The amount of type layouts in this range.
86        #[inline]
87        pub const fn len(&self) -> usize {
88            (self.bits0 & Self::LEN_MASK) as usize
89        }
90
91        /// Whether this range of type layouts is empty.
92        #[inline]
93        pub const fn is_empty(&self) -> bool {
94            self.len() == 0
95        }
96    }
97
98

100
101)}
