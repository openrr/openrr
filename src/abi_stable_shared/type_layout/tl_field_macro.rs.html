2#[macro_export]
3macro_rules! declare_comp_tl_field {(
4    attrs=[ $($extra_attrs:meta),* $(,)* ]
5) => (
6
7    /// A `TLField` represented as a `u64`,
8    /// expadable to a `TLField` by calling the `expand` method.
9    #[repr(transparent)]
10    #[derive(Debug, Copy, Clone)]
11    $(#[ $extra_attrs ])*
12    pub struct CompTLField{
13        bits0:u64,
14    }
15
16    /// The underlying representation of `CompTLField`.
17    pub type CompTLFieldRepr=u64;
18
19    impl CompTLField{
20        const NAME_OFFSET:u32=0;
21
22        const LIFETIME_INDICES_OFFSET:u32=StartLen::BIT_SIZE;
23
24        const FIELD_ACCESSOR_OFFSET:u32=Self::LIFETIME_INDICES_OFFSET+LifetimeRange::BIT_SIZE;
25
26        const TYPE_LAYOUT_OFFSET:u32=Self::FIELD_ACCESSOR_OFFSET+CompFieldAccessor::BIT_SIZE;
27        const TYPE_LAYOUT_SR_MASK:u64=TypeLayoutIndex::MASK as u64;
28
29        const IS_FUNCTION_OFFSET:u32=Self::TYPE_LAYOUT_OFFSET+TypeLayoutIndex::BIT_SIZE;
30        const IS_FUNCTION_BIT_SIZE:u32=1;
31
32        /// The amount of bits necessary to represent a CompTLField.
33        pub const BIT_SIZE:u32=Self::IS_FUNCTION_OFFSET+Self::IS_FUNCTION_BIT_SIZE;
34
35        /// Constructs a CompTLField.
36        #[inline]
37        pub const fn new(
38            name:StartLen,
39            lifetime_indices:LifetimeRange,
40            field_accessor:CompFieldAccessor,
41            layout: TypeLayoutIndex,
42            is_function:bool,
43        )->Self{
44            // A compile-time assertion that the bit fields fit inside a u64
45            let _:[(); (64-Self::BIT_SIZE)as usize];
46
47            let bits0={
48                 ((name.to_u26() as u64)<<Self::NAME_OFFSET)
49                |((lifetime_indices.to_u21() as u64)<<Self::LIFETIME_INDICES_OFFSET)
50                |((field_accessor.to_u3() as u64)<<Self::FIELD_ACCESSOR_OFFSET)
51                |((layout.to_u10() as u64)<<Self::TYPE_LAYOUT_OFFSET)
52                |((is_function as u64)<<Self::IS_FUNCTION_OFFSET)
53            };
54
55            CompTLField{bits0}
56        }
57
58        /// Gets the range representing the name in the string slice field inside
59        /// the `SharedVars` field of the `TypeLayout` that contains this.
60        #[inline]
61        pub const fn name_start_len(&self)->StartLen{
62            StartLen::from_u26((self.bits0>>Self::NAME_OFFSET) as u32)
63        }
64
65        /// Gets the index of the type layout of the field in
66        /// the slice of type layouts inside
67        /// the `SharedVars` field of the `TypeLayout` that contains this.
68        #[inline]
69        pub const fn type_layout_index(&self)-> usize {
70            ((self.bits0>>Self::TYPE_LAYOUT_OFFSET)&Self::TYPE_LAYOUT_SR_MASK)as usize
71        }
72
73
74        #[inline]
75        const fn lifetime_indices_bits(&self)-> u32 {
76            (self.bits0>>Self::LIFETIME_INDICES_OFFSET)as u32
77        }
78
79        /// Whether this field is a function.
80        /// This is only true if the type is a function pointer(not inside some other type).
81        #[inline]
82        pub const fn is_function(&self)->bool{
83            (self.bits0 & (1<<Self::IS_FUNCTION_OFFSET))!=0
84        }
85
86        #[inline]
87        pub(crate) const fn std_field(
88            name:StartLen,
89            lifetime_indices:LifetimeRange,
90            layout: u16,
91        )->Self{
92            Self::new(
93                name,
94                lifetime_indices,
95                CompFieldAccessor::DIRECT,
96                TypeLayoutIndex::from_u10(layout),
97                false,
98            )
99        }
100    }
101)}
