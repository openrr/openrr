7/// An accessible object role.
8/// To think of it in terms of HTML, any semantic element likely has a corollary in this enum.
9/// For example: `<button>`, `<input>`, `<form>` or `<h4>`.
10/// Non-semantic elements like `<span>`, `<div>` and `<b>` will not be represented here, and this information is not passed through via the atspi library.
11/// TODO: add examples for GTK/Qt libraries in addition to HTML examples.
12#[repr(u32)]
13pub enum Role {

22	/// Object is a 2d directional indicator.
23	Arrow,
24	/// Object contains one or more dates, usually arranged into a 2d list.

26	/// Object that can be drawn into and is used to trap events.
27	Canvas,
28	/// A choice that can be checked or unchecked and provides a separate indicator for the current state.
29	CheckBox,
30	/// A menu item that behaves like a check box. See [`Self::CheckBox`].
31	CheckMenuItem,
32	/// A specialized dialog that lets the user choose a color.
33	ColorChooser,
34	/// The header for a column of data.
35	ColumnHeader,
36	/// A list of choices the user can select from.
37	ComboBox,
38	/// An object which allows entry of a date.
39	DateEditor,

41	DesktopIcon,

43	DesktopFrame,

620pub mod tests {
621	use super::Role;
622	use zvariant::serialized::Context;
623	use zvariant::{to_bytes, LE};
624
625	const HIGHEST_ROLE_VALUE: u32 = 129;
626
627	#[test]
628	fn test_serialization_matches_from_impl() {
629		let ctxt = Context::new_dbus(LE, 0);
630
631		for role_num in 1..=HIGHEST_ROLE_VALUE {
632			let from_role = Role::try_from(role_num)
633				.unwrap_or_else(|_| panic!("Unable to convert {role_num} into Role"));
634			let encoded = to_bytes(ctxt, &from_role)
635				.unwrap_or_else(|_| panic!("Unable to encode {from_role}"));
636			println!("ENCODED: {encoded:?}");
637
638			let (zbus_role, _) = encoded.deserialize().expect("Unable to decode {encoded:?}");
639
640			assert_eq!(from_role, zbus_role, "The serde `Data::deserialize` and `From<u32>` impls produced different results. The number used was {role_num}, it produced a Role of {from_role}, but the from_slice(...) implementation produced {zbus_role}");
641			assert_eq!(
642				from_role as u32, role_num,
643				"The role number {role_num} does not match the representation of the role {}",
644				from_role as u32
645			);
646		}
647	}
