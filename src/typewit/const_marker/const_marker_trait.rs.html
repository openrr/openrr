1/// A type that represents a constant
2/// 
3/// # Example
4/// 
5/// Emulating generic associated constants on stable
6/// 
7/// (this example Requires Rust 1.65.0)
8/// 
9#[cfg_attr(not(feature = "rust_1_65"), doc = "```ignore")]
10#[cfg_attr(feature = "rust_1_65", doc = "```rust")]
11/// use typewit::const_marker::{ConstMarker, ConstMarkerOf};
12/// 
13/// 
14/// assert_eq!(make_array::<u8>(), ([3], [3, 3, 3]));
15/// 
16/// assert_eq!(make_array::<&str>(), (["hi"], ["hi", "hi", "hi"]));
17/// 
18/// const fn make_array<T: MakeArray>() -> ([T; 1], [T; 3]) {
19///     (T::Make::<1>::VAL, T::Make::<3>::VAL)
20/// }
21/// 
22/// // `MakeArray` can make arrays of any length without writing a bound for each length!!
23/// trait MakeArray: Sized {
24///     // emulates `const Make<const N: usize>: [Self; N];` (a generic associated constant)
25///     type Make<const N: usize>: ConstMarkerOf<[Self; N]>;
26/// }
27/// 
28/// 
29/// impl MakeArray for u8 {
30///     type Make<const N: usize> = MakeArrayConst<Self, N>;
31/// }
32///
33/// impl<'a> MakeArray for &'a str {
34///     type Make<const N: usize> = MakeArrayConst<Self, N>;
35/// }
36/// 
37/// struct MakeArrayConst<T, const N: usize>(core::marker::PhantomData<T>);
38/// 
39/// impl<const N: usize> ConstMarker for MakeArrayConst<u8, N> {
40///     type Of = [u8; N];
41///     const VAL: Self::Of = [3; N];
42/// }
43/// 
44/// impl<'a, const N: usize> ConstMarker for MakeArrayConst<&'a str, N> {
45///     type Of = [&'a str; N];
46///     const VAL: Self::Of = ["hi"; N];
47/// }
48/// 
49/// ```
50/// 
51pub trait ConstMarker: Sized {
52    /// The type of this constant
53    type Of;
54
55    /// The value of this constant
56    const VAL: Self::Of;
57}
58
59
60/// Trait alias for `ConstMarker<Of = Of>`
61/// 
62/// This trait also shows the `Of` type in unsatisfied trait bound errors
63/// (`ConstMarker<Of = ...>` bounds obfuscate this a bit as of 1.89.0) 
64#[cfg_attr(feature = "rust_1_83", diagnostic::on_unimplemented(
65    message = "`{Self}` is not a type-level constant of type `{Of}`",
66))]
67pub trait ConstMarkerOf<Of>: ConstMarker<Of = Of> {}
68
69impl<C, Of> ConstMarkerOf<Of> for C
70where
71    C: ConstMarker<Of = Of>,
72{}
