Struct RBox

Source 
#[repr(C)]
pub struct RBox<T> { /* private fields */ }
Expand description

Ffi-safe equivalent of std::box::Box.

ยงExample

Declaring a recursive datatype.

use abi_stable::{
    std_types::{RBox, RString},
    StableAbi,
};

#[repr(u8)]
#[derive(StableAbi)]
enum Command {
    SendProduct {
        id: u64,
    },
    GoProtest {
        cause: RString,
        place: RString,
    },
    SendComplaint {
        cause: RString,
        website: RString,
    },
    WithMetadata {
        command: RBox<Command>,
        metadata: RString,
    },
}
