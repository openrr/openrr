fn mode(&mut self, mode: u32) -> &mut Self

Sets the mode to create new directories with.

This option defaults to 0o777.

§Examples
use async_fs::{DirBuilder, unix::DirBuilderExt};

let mut builder = DirBuilder::new();
builder.mode(0o755);
